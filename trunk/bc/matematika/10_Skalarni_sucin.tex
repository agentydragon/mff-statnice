\subsection{Vlastnosti v reálném i komplexním pøípadì}
\begin{e}{Definice}{0}{0}
Nech $V$ je vektorovı prostor nad $\mathbb{C}$. Potom zobrazení (funkce) z kartézského souèinu $V \times V \rightarrow \mathbb{C}$, které dvojici vektorù $x$ a $y$ pøiradí èíslo $\left<x,y\right>$ se nazıvá \emph{skalární souèin}, pokud splòuje následující axiomy (pro všechny $x, x', y \in V$ a $\alpha, \beta \in \mathbb{C}$):
\begin{penumerate}
	\item $\left<x,x\right> \ge 0, \,\, \left<x,x\right> =0 \Leftrightarrow x = 0 \hfill \textit{(positivní definitnost)}$
	\item $\left<\alpha x+\beta x', y\right> = \alpha \left<x,y\right> + \beta \left<x',y\right> \hfill \textit{(bilinearita)}$
		\begin{penumerate}
			\item $\left<\alpha x,y\right> = \alpha \left<x,y\right>$
			\item $\left<x+x',y\right> = \left<x,y\right> + \left<x',y\right>$
		\end{penumerate}
	\item $\left<x,y\right> = \overline{\left<y,x\right>} \hfill \textit{(symetrie - komplexnì sdruené)}$ \\
		\begin{e}{Poznámka}{0}{0}
			Pro $V'$ nad $\mathbb{R}$ a vektory $\forall x,y \in V'$: $\left<x,y\right>=\left<y,x\right>$
		\end{e}
\end{penumerate}

Skalární souèin znaèíme: $\left<x,y\right>$, $\left<x|y\right>$, $x.y$ \dots
\end{e}

\begin{e}{Pozorování}{0}{0}
\begin{pitemize}
	\item $\left<x,x\right>=\overline{\left<x,x\right>}$, tedy je nutnì reálné ($\in\mathbb{R}$) i pro skalární souèiny nad $\mathbb{C}$
	\item $\left<x,\alpha y\right> = \overline{\left<\alpha y,x\right>} = \overline{\alpha}.\overline{\left<y,x\right>} = \overline{\alpha}.\left<x,y\right>$
	\item Skalární souèin mùe nabıvat zápornıch hodnot
\end{pitemize}
\end{e}

\begin{e}{Definice}{0}{0}
Ekvivalentní definice: \emph{Skalární souèin} je pozitivnì definitní (1) bilineární forma (2). V $\mathbb{R}$ navíc symetrická (3). V $\mathbb{C}$ navíc forma, její matice je hermitovská (3).
\end{e}

\begin{e}{Pøíklady}{0}{0}
\begin{pitemize}
	\item \uv{Standardní} skalární souèin pro $\mathbb{C}^n, \mathbb{R}^n$: $$\left<x,y\right> = \sum_{i=1}^{n} x_i \overline{y_i}$$
	\item Jinı souèin v $\mathbb{R}^n$ definovanı pomocí regulární matice $A$ øádu $n$
		$$\left<x,y\right>=x^T A^T A y \;\;\;\;\;\; (\textit{pozorování: } \left<x,x\right> = x^T A^T A x = \sum_{i=1}^n (Ax)_i^2)$$
	\item Skalární souèin ve vektorovém prostoru $C[a,b]$ (integrovatelnıch funkcí na intervalu $[a,b]$):
		$$\left<f,g\right> = \int_a^b f(x) g(x) dx$$
\end{pitemize}
\end{e}

\subsection{Norma}

\begin{e}{Definice}{0}{Norma}
\emph{Norma} na vektorovém prostoru V (nad $\mathbb{R}$ nebo nad $\mathbb{C}$) je zobrazení $V \rightarrow \mathbb{R}$, které pøiradí vektoru $x \in V$ èíslo $\|x\|$ a splòuje axiomy:
\begin{penumerate}
	\item $\forall x \in V: \|x\| \ge 0, \|x\|=0 \Leftrightarrow x = 0$
	\item $\forall x \in V, \forall \alpha \in \mathbb{C} (\mathbb{R}): \|\alpha x\| = |\alpha|.\|x\|$
	\item $\forall x,y \in V: \|x\| \ge 0, \|x+y\| \le \|x\| + \|y\| \hfill \textit{(trojúhelníková nerovnost)}$
\end{penumerate}
Norma $\|x\|$ má vıznam \uv{délky} vektoru $x$.
\end{e}

\begin{e}{Definice}{0}{Normovanı vekt. prostor}
Vektorovı prostor s nìjakou normou nazıváme \emph{normovanı}.
\end{e}

\begin{e}{Pøíklady}{0}{0}
\begin{pitemize}
\item Norma urèená skalárním souèinem
	$$\|x\| = \sqrt{\left<x,x\right>}$$
	\begin{e}{Dùkaz}{0}{0}
		(1), (2) plyne z axiomù skalárního souèinu, (3):
		\begin{align*}
			\left<x,y\right>^2 \le \left<x,x\right>\left<y,y\right> & \Rightarrow & \left<x,y\right> & \le & \sqrt{\left<x,x\right> \left<y,y\right>} \\
			& \Leftrightarrow & \left<x,x\right>+\left<y,y\right>+2\left<x,y\right> & \le & (\sqrt{\left<x,x\right>} + \sqrt{\left<y,y\right>})^2 \\
			& \Leftrightarrow & \left<x+y,x+y\right> & \le & (\sqrt{\left<x,x\right>} + \sqrt{\left<y,y\right>})^2 \\
			& \Leftrightarrow & \|x+y\| & \le & \|x\| + \|y\|
		\end{align*}
		Kde první nerovnost je dùsledek Cauchy-Swarzovy nerovnosti...

		Ze standardního skalárního souèinu na $\mathbb{R}^n$ dostaneme euklidovskou normu (tj. \uv{délku} vektoru podle Pythagorovy vìty) a euklidovskou vzdálenost (vzdálenost bodù $u$ a $v$ je $\|u-v\|$). Kadı vektorovı prostor se skalárním souèinem $\left<.,.\right>$ je normovanım vektorovım prostorem ($\|x\|=\sqrt{\left<x,x\right>}$), tedy i metrickım prostorem ($d(x,y)=\|x-y\|$) a tedy i topologickım prostorem.
	\end{e}
\item $L_1$ norma na $\mathbb{R}^n$:
	$$\|x\| = \sum_{i=1}^{n} |x_i|$$
\item $L_2$ norma na $\mathbb{C}^n$ - Euklidovská norma:
	$$\|x\| = \sqrt{\sum_{i=1}^{n} x_i \overline{x_i}}$$
\item $L_p$ norma na $\mathbb{R}^n$:
	$$\|x\| = \sqrt[p]{\sum_{i=1}^{n} |x_i|^p}$$
\item $L_{\infty}$ norma (stejnì jako $L_1$ norma neodpovídá ádnému skalárnímu souèinu):
	$$\|x\| = \max_{i=1,\dots,n} \left( |x_i| \right)$$
\item Norma v prostoru integrovatelnıch funkcí na intervalu $[a,b]$ - $C[a,b]$
	$$\|f(x)\| = \int_a^b f^2(x) dx$$
\end{pitemize}
\end{e}


\subsection{Cauchy-Schwarzova nerovnost}


\begin{e}{Vìta}{0}{Cauchyho-Schwarzova nerovnost}
Nech $V$ je prostor se skalárním souèinem nad $\mathbb{C}$ a $\|x\|$ je norma odvozená ze skalárního souèinu. Potom platí:
$$\left|\left<x,y\right>\right| \le \|x\|\cdot\|y\| \;\;\;\;\;\; (\forall x,y \in V)$$

\begin{e}{Dùkaz}{0}{0}
Pro $x=0$ nebo $y=0$ máme $0 \le 0$.

Pro libovolné $\alpha \in \mathbb{C}$ platí $\|x+\alpha y\|^2 \ge 0$ (platí i bez $( )^2$)
\begin{align*}
\|x+\alpha y\|^2 = \left<x+\alpha y, x+\alpha y\right> = \left<x, x+\alpha y\right> + \alpha \left<y, x+\alpha y\right> = \\
= \left<x,x\right> + \overline{\alpha}\left<x,y\right> + \alpha\left<y,x\right> + \alpha \overline{\alpha}\left<y,y\right>
\end{align*}
Zvolíme $\alpha = \frac{-\left<x,y\right>}{\left<y,y\right>}$ (\emph{tím se eliminují $\overline{\alpha}\left<x,y\right>$ a $\alpha \overline{\alpha}\left<y,y\right>$})

Po dosazení:
\begin{align*}
0 &\le& \left<x,x\right> + \alpha \left<y,x\right> \\
0 &\le& \left<x,x\right> - \frac{\left<x,y\right>}{\left<y,y\right>}\left<y,x\right> \\
\left<x,y\right>.\left<y,x\right> &\le& \left<x,x\right>.\left<y,y\right> \\
|\left<x,y\right>|^2 & \le & \|x\|^2 . \|y\|^2 \\
\textit{...a po odmocnìní} \\
|\left<x,y\right>| & \le & \|x\|.\|y\|
\end{align*}

\end{e}

\begin{obecne}{Druhı monı dùkaz}
\noindent Nadefinujeme promìnnou $t\in\mathbb{R}$ a zavedeme funkci
$$p(t):=\left<u+t\cdot v,u+t\cdot v\right>=||u+tv||^2$$

\noindent
Víme: $p(t)\geq 0\ \forall t\in\mathbb{R}$ (z axiomu 1 skal. souèinu). Z linearity plyne, e $\left<u+tv,u+tv\right>=\left<u,u+tv\right>+t\left<v,u+tv\right>=\left<u,u\right>+t\left<u,v\right>+t\left<v,u\right>+t^2\left<v,v\right>=||u||^2+2t\left<u,v\right>+t^2||v||^2$. Tj. dostáváme $p(t)$ jako kvadratickou funkci promìnné $t$:
$$p(t)=t^2||v||^2+2t\left<u,v\right>+||u||^2$$

\noindent
Protoe $p(t)$ má nezáporné hodnoty na celém $\mathbb{R}$, musí mít tato rovnice max. jedno øešení, tj. diskriminant pøi poèítání koøenù nesmí bıt kladnı:
$$D=b^2-4ac=4\left<u,v\right>^2-4||u||^2||v||^2\leq 0$$

\noindent
Po vydìlení ètyømi a odmocnìní dostáváme:
$$|\left<u,v\right>|\leq ||u||\cdot||v||$$
\end{obecne}

\begin{e}{Dùsledek}{0}{0}
Platnost trojúhelníkové nerovnosti pro normy odvozené od skalárního souèinu -- tj. normy odvozené od skalárního souèinu splòují všechny axiomy normy.
\end{e}

\begin{e}{Dùsledek}{0}{0}
Nech $x=(x_1,x_2,\dots,x_n)^T$, $y=(1,1,\dots,1)^T$ jsou dva vektory, pak pro standardní skalární souèin platí
\begin{align*}
|\left<x,y\right>| &= \sum_{i=1}^n x_i\cdot 1\\
\|x\| &= \sqrt{\sum_{i=1}^n x_i^2}\\
\|y\| &= \sqrt{n}
\end{align*}
po dosazení do Cauchy-Schwarzovy nerovnosti okamitì dostaneme nerovnost mezi aritmetickım a kvadratickım prùmìrem
$$\frac{1}{n}\sum_{i=1}^{n} x_i \le \sqrt{\frac{1}{n}\sum_{i=1}^{n} x_i^2}$$
\end{e}



\begin{e}{Dùsledek}{0}{0}
Ve vektorovıch prostorech nad $\mathbb{R}$ a $\mathbb{C}$ lze definovat \emph{úhel}, svíranı dvìma vektory: 
$$\cos\varphi = \frac{\left<u,v\right>}{\|u\|\ \|v\|}$$
a Cauchyho-Schwarzova nerovnost zaruèuje, e $|\cos\varphi|\leq 1$.
\end{e}

\begin{e}{Dùsledek}{0}{0}

$$\|u-v\|^2=\|u\|^2+\|v\|^2-2\|u\|\ \|v\|\cos\varphi$$
\end{e}
\end{e}


\subsection{Kolmost}
\begin{e}{Definice}{0}{kolmé vektory}
Vektory $x$ a $y$ z prostoru se skalárním souèinem jsou vzájemnì \emph{kolmé} (\emph{ortogonální}), pokud $\left<x,y\right>=0$, znaèíme $x \bot y$.
\end{e}

\begin{e}{Definice}{0}{ortogonální a ortonormální systém}
Soustava (systém) vektorù $v_1, \dots, v_n$ se nazıvá \emph{ortogonální}, jestlie $\left<v_i,v_j\right> = 0$ ($v_i \bot v_j$) pro $\forall i \neq j$ (tj. všechny její vektory jsou navzájem kolmé). 

\noindent
Platí-li ještì navíc $\|v_i\|=1$ pro $\forall i=1,\dots,n$, jedná se o soustavu \emph{ortonormální} (vektory jsou kolmé a navíc mají jednotkovou normu).
\end{e}

\begin{e}{Pozorování}{0}{0}
Kadı systém nenulovıch vzájemnì kolmıch vektorù (tj. i ortonormální nebo ortogonální) je lineárnì nezávislı.
\end{e}

\begin{e}{Dùsledek}{0}{0}
Jestlie ortogonální systém generuje celı vektorovı prostor, je jeho bází.
\end{e}

\begin{e}{Algoritmus}{0}{Gram-Schmidtova ortogonalizace}
Tento algoritmus zajišuje pøevedení libovolné báze $(v_1,\dots,v_n)$ vektorového prostoru $V$ na ekvivalentní ortogonální bázi $(w_1,\dots,w_n)$. Ortonormalizace báze u po jeho probìhnutí znamená jen vynásobení kadého $w_i$ èíslem $\frac{1}{\|w_i\|}$. Jeho prùbìh:
\begin{penumerate}
    \item Zvolme $w_1 := v_1$.
    \item Pro $i$ postupnì od $1$ do $n$ opakujme:
\par\noindent
    Najdi $w_i=v_i-a_{i,1}w_1 - a_{i,2}w_2 - \dots - a_{i,i-1}w_{i-1}$ tak, aby pro $\forall j\in\{1,\dots,i\}$ platilo:
    $$w_i\bot w_j$$
    Dá se ukázat e koeficienty $a_{i,j}$ jsou tvaru
    $$a_{i,j}=\frac{\left<v_i,w_j\right>}{\|w_j\|^2}$$
    \item Po $n$ iteracích dostaneme $w_1,\dots,w_n$ jako ortogonální bázi prostoru $V$.
\end{penumerate}

\textbf{Alternativní postup - Gram-Schmidtova normalizace}:
\begin{penumerate}
	\item Dány: $x_1,\dots,x_m \in V$ lineárnì nezávislé.
	\item Pro $k=1,\dots,m$ proveï:
		$$y_k := x_k - \sum_{j=1}^{k-1} \left<x_k,z_j\right>z_j$$
		$$z_k := \frac{1}{\|y_k\|}y_k$$
	\item Ukonèi: $z_1,\dots,z_m$ je ortonormální systém ve $V$ a\\$\mathcal{L}(z_1,\dots,z_m) = \mathcal{L}(x_1,\dots,x_m)$
\end{penumerate}
\end{e}

\begin{e}{Dùsledek}{0}{0}
Buï $(v_1,\dots,v_n)$ báze vekt. prostoru se skal. souèinem. Potom existuje ortonormální báze $(w_1,\dots,w_n)$, kdy pro kadé $k\in\{1,\dots,n\}$ je $\mathcal{L}(v_1,\dots,v_k)=\mathcal{L}(w_1,\dots,w_k)$. Díky tomu se kadı ortogonální systém vektorù v koneènìdimensionálním vekt. prostoru se skalárním souèinem dá rozšíøit na ortogonální bázi (to mùeme díky Gram-Schmidtovì ortogonalizaci a Steinitzovì vìtì o vımìnì).
\end{e}

\begin{e}{Vìta}{0}{Fourierovy koeficienty}
Máme-li danou nìjakou ortonormální bázi $B=b_1,\dots,b_n$ vektorového prostoru $V$, pak pro kadé $x\in V$ platí:
$$x=\sum_{i=1}^n\left<x,b_i\right>b_i$$
a souøadnice $\left<x,b_i\right>$ nazveme \emph{Fourierovy koeficienty} vektoru $x$.
\end{e}

\begin{e}{Poznámka}{0}{0}

\subsection{Ortogonální doplnìk a jeho vlastnosti}
\begin{e}{Definice}{0}{0}
Nech $V$ je mnoina vektorù ve vektorovém prostoru $W$ se skalárním souèinem. \emph{Ortogonálním doplòkem} $V$ (znaèíme $V^{\bot}$) rozumíme mnoinu
$$V^{\bot}=\left\{v \in W; \forall x \in V: \left<v,x\right> = 0 \right\}$$
\end{e}

\begin{lemmaN}{Vlastnosti}
Nech $V$ je podprostor prostoru $W$ koneèné dimenze. Potom platí:
\begin{penumerate}
	\item $V^{\bot}$ je podprostor $W$
	\item $\dim(V^{\bot})=\dim(W)-\dim(V)$
	\item $(V^{\bot})^{\bot} = V \hfill \textit{(z rozšiøitelnosti ortogonální báze)}$ 
	\item $V \cap V^{\bot} = \{0\}, \;\;\; V \oplus V^{\bot} = W \\ \textit{(operace} \oplus \textit{je spojení dvou podprostorù...} \mathcal{L}(V \cup V^{\bot})\textit{)}$
	\item $U,V$ podprostory $W$. Je-li $U \subseteq V$, pak $U^{\bot} \supseteq V^{\bot}$\\
		$(x \in V^{\bot} \Leftrightarrow x \bot y \in V \Rightarrow x \bot u \in U \Leftrightarrow x \in U^{\bot})$
	\item $(U \cap V)^{\bot} = U^{\bot} \oplus V^{\bot}$
	\item $(U \oplus V)^{\bot} = U^{\bot} \cap V^{\bot}$
\end{penumerate}
\end{lemmaN}

\begin{e}{Definice}{0}{Ortogonální projekce}
\emph{Ortogonální projekce} vekt. prostoru $V$ na podprostor $U\subset V$ je zobrazení, které kadému vektoru $v\in V$ pøiøadí vektor $u\in U$ tak, e 
$$\|v-u\|=\min\{\|v-w\|, w \in U\}$$
tedy vektor $u \in U$, kterı má ze všech vektorù z $U$ nejmenší vzdálenost od $v$. Ten se pak nazıvá \emph{ortogonální projekcí} vektoru $v$.
\end{e}
