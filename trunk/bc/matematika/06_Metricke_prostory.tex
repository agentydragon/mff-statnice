\subsubsection*{Metrický prostor, metrika}

\begin{e}{Definice}{0}{metrický prostor}
\emph{Metrický prostor} je dvojice $(M, d)$, kde $M$ je množina a $d: M \times M \rightarrow \mathbb{R}$ je zobrazení, zvané \emph{metrika}, splòující tøi axiomy:
\begin{penumerate}
    \item $d(x, y) = 0 \Leftrightarrow x = y$
    \item $d(x, y) = d(y, x)$ \hfill \emph{(symetrie)}
    \item $d(x, y) \le d(x, z) + d(z, y)$ \hfill\emph{(trojúhelníková nerovnost)}
\end{penumerate}
Metrické prostory jsou abstrakcí jevu vzdálenosti. Z axiomù 1. a 3. vyplývá nezápornost hodnot metriky (která se ale vìtšinou explicitnì uvádí jako souèást prvního axiomu). Prvky metrického prostoru nazveme \emph{body}.
\end{e}

\begin{obecne}{Pøíklady metrik}
\begin{itemize}
    \item Nech $M = \mathbb{R}^n$ a $p \ge 1$ je reálné èíslo. Na $M$ definujeme metriky\\(kde $x = (x_1, x_2, \dots, x_n)$, $y = (y_1, y_2, \dots , y_n)$)
$$d_p(x, y) = \left(\sum_{i=1}^{n}|x_i - y_i|^p\right)^{1/p}$$

\noindent Potom:\begin{pitemize}
\item Pro $p = 1, n = 1$ dostáváme metriku $|x - y|$.
\item Pro $p = 2, n \ge 2$ dostáváme euklidovskou metriku
$$d_2(x, y) = \parallel x - y \parallel = \sqrt{(x_1-y_1)^2+(x_2-y_2)^2+\dots+(x_n-y_n)^2}$$
\item Pro $p = 1, n \ge 2$ dostáváme tzv. pošáckou metriku a pro $p \rightarrow \infty$ maximovou
metriku
$$d_1(x, y) = \max_{1 \le i \le n} |x_i - y_i|$$
\end{pitemize}
    \item
Buï $X$ libovolná množina. $F(X)$ oznaème množinu všech omezených funkcí $f:X\to\mathbb{R}$ a definujme funkci $d$ pøedpisem
$$d(f,g)=\sup_{x\in X}|f(x)-g(x)|$$
Pak $(F(X),d)$ je metrický prostor (s tzv. supremovou metrikou).
    \item
Vezmeme $M = \mathcal{C}[a, b]$ (množina reálných funkcí definovaných a spojitých an intervalu $[a, b]$) a reálné èíslo $p \geq 1$. Podobnì jako v prvním pøíkladu máme na $M$ metriky
$$d_p\left(f, g\right) = \left( \int_a^b \lvert f(x) - g(x) \rvert^p \; \mathrm{d}x\right)^{1/p}.$$
Pro $p = 1$ máme integrální metriku a $p \to \infty$ dává maximovou metriku
z druhého pøíkladu. Vezmeme-li širší tøídu funkcí $M = \mathcal{R}[a, b]$ (funkce mající na $[a, b]$ Riemannùv
integrál), je $d_p(f, g)$ definovaná, ale nesplòuje axiom 1) a nedostáváme metriku.
Zmìníme-li napøíklad hodnotu funkce $f \in \mathcal{R}[a, b]$ v jediném bodì, dostaneme
odlišnou funkci $f_0 \in \mathcal{R}[a, b]$, ale $d_p(f, f_0) = 0$. (Tato potíž se odstraní tak, že
místo s $\mathcal{R}[a, b]$ se pracuje s $\mathcal{R}[a, b]/\sim$ pro vhodnou relaci ekvivalence $\sim$.)
    \item
Na souvislém grafu $G = (M, E)$ s množinou vrcholù $M$ máme metriku $d(u, v)$ odpovídající poètu hran na nejkratší cestì v $G$ spojující $u$ a $v$.
    \item
Je-li $A$ koneèná množina (abeceda), máme na množinì $M = A^m$ slov délky $m$ nad abecedou $A$ tzv. Hammingovu metriku ($u = a_1 a_2 \ldots a_m, v = b_1 b_2 \ldots b_m$), kde $d(u, v)$ odpovídá poètu souøadnic $i$, pro nìž $a_i \neq b_i$. Mìøí míru odlišnosti obou slov -- jaký nejmenší poèet zmìn v písmenech staèí k pøemìnì $u$ ve $v$.
    \item
Úplnì triviální pøíklad metriky dostaneme, když na nìjaké množinì $M$ položíme $d(x,y)=1$ pro $x\neq y$ a $d(x,x)=0$.
\end{itemize}
\end{obecne}

\begin{e}{Definice}{0}{euklidovský prostor}
\emph{Euklidovským prostorem} rozumíme metrický prostor $(\mathbb{R}^n, d_2)$, kde $d_2$ je funkce daná pøedpisem $d_2(x,y)=\sqrt{\sum_{i=1}^n(x_i-y_i)^2}$.
\end{e}

\subsubsection*{Konvergence}

\begin{e}{Definice}{0}{konvergence posloupnosti bodù}
Øekneme, že posloupnost bodù $(x_n)_{n\geq 0}$ nìjakého metrického prostoru $(M, d)$ \emph{konverguje k bodu $x \in M$} ($x_n\to x$), nebo že $x=\lim_{n\geq 0} x_n$, jestliže
$$\forall\varepsilon>0\ \exists n_0:n\geq n_0\;\Rightarrow\; d(x_n,x)<\varepsilon.$$
\end{e}

\begin{e}{Poznámka}{0}{vlastnosti konvergence}
Nech je dán metrický prostor $(M, d)$ a v nìm posloupnost bodù $(x_n)_{n\geq 0}$. Potom platí:
\begin{penumerate}
    \item Jestliže pro bod $y\in M$ platí, že $\exists n_0\in\mathbb{N}: \forall n\geq n_0: x_n = y$, pak $x_n\to y$
    \item Nech $x_n\to y_1$ a zároveò $x_n\to y_2$. Potom $y_1=y_2$.
    \item Podposloupnost konvergentní posloupnosti konverguje ke stejnému bodu.
\end{penumerate}
\begin{e}{Dùkaz}{0}{0}
Obdobnì jako pro konvergenci posloupností z $\mathbb{R}$ -- pouze místo absolutní hodnoty použijeme obecnou metriku.
\end{e}
\end{e}
