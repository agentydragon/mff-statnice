\subsection{Spojitost a stejnomìrná spojitost}

\subsubsection*{Otevøené a uzavøené množiny}

\obrazeknahore{matematika/obrazky/prostory_koule}{Rùzné koule}{fig:koule}

\begin{definiceN}{otevøená a uzavøená koule}
Buï $(M,d)$ metrický prostor, $x \in M, r > 0$, potom
\begin{pitemize}
\item \emph{otevøenou koulí} (\emph{$r$-okolím}) se støedem $x$ a polomìrem $r$ nazveme množinu
$$B(x,r)=\{y \in M | d(x,y)<r\}$$
\item \emph{uzavøenou koulí} (\emph{$r$-okolím}) se støedem $x$ a polomìrem $r$ nazveme množinu
$$\overline{B}(x,r) = \{y \in M | d(x,y) \le r \}$$
\end{pitemize}
\end{definiceN}

\begin{poznamka}
    Na obrázku \ref{fig:koule} je ilustrace nìkolika koulí na rùzných metrikách. Obrázky jsou z~\url{http://kma.me.sweb.cz/metr-prostor.pdf}.
\end{poznamka}


\begin{definiceN}{otevøená a uzavøená množina}
Buï $(M,d)$ metrický prostor, $G \subseteq M$, potom
\begin{pitemize}
\item $G$ je \emph{otevøená} v $M$, pokud
$$\forall x \in G\;\exists r >\;0:\; B(x,r)\subseteq G$$
(tj. množina $G$ je okolím každého svého bodu),
\item $G$ je \emph{uzavøená} v $M$, pokud její doplnìk $M \setminus G$ je otevøený v $M$.
\end{pitemize}
Otevøená koule je otevøená množina v každém metrickém prostoru. Podobnì pro uzavøenou kouli.
\end{definiceN}

\begin{vetaN}{vlastnosti otevøených množin}
Buï $(M,d)$ metrický prostor, potom platí, že
\begin{penumerate}
\item $\emptyset$, $M$ jsou otevøené v $M$,
\item koneèný prùnik otevøených množin je otevøená množina v $M$,
\item libovolnì velké sjednocení otevøených množin je otevøená množina v $M$.
\end{penumerate}
\begin{ideadukazu}
Èást 1. platí pro prázdnou množinu (podmínka je prázdná) a doplòkem i pro celé $M$. Proè v èásti 2. musí být prùnik koneèný? Vezmìme nekoneènou posloupnost otevøených koulí $\left(B_n\left(x_0, 1/n\right)\right)_{n \geq 1}$, pak jejich prùnikem je pouze $\lbrace x_0\rbrace$, což není otevøená množina!
\end{ideadukazu}
\end{vetaN}

\begin{vetaN}{vlastnosti uzavøených množin}
Nech $(M,d)$ je metrický prostor, potom platí, že
\begin{penumerate}
\item $\emptyset$, $M$ jsou uzavøené v $M$,
\item libovolný prùnik uzavøených množin je uzavøená množina v $M$,
\item koneèné sjednocení uzavøených množin je uzavøená množina v $M$.
\end{penumerate}
\end{vetaN}
\begin{definiceN}{okolí bodu, vnitøní, vnìjší a další body}
Buï $(M, d)$ metrický prostor, $a \in M$ jeho bod. Každou otevøenou množinu $U \subset M$ splòující $a \in U$ nazveme \emph{okolím bodu $a$}. Tedy každá koule $B(a, r)$ je okolím bodu $a$ a ovšem každé okolí
bodu $a$ obsahuje nìjakou takovou kouli. V následujících definicích buïte $a \in M$ bod, $X \subset M$ množina a $U$ okolí bodu $a$. Øekneme, že
\begin{pitemize}
    \item $a$ je \emph{vnitøním} bodem $X$, když existuje $U$ tak, že $U \subset X$,
    \item $a$ je \emph{vnìjším} bodem $X$, když existuje $U$ tak, že $U \subset M\setminus X$,
    \item $a$ je \emph{hranièním} bodem $X$, když každé $U$ protíná $X$ i $M\setminus X$,
    \item $a$ je \emph{limitním} bodem $X$, když je pro každé $U$ prùnik $U \cap X$ nekoneèný,
    \item $a$ je \emph{izolovaným} bodem $X$, když existuje $U$ tak, že $U \cap X = \lbrace a\rbrace$.
\end{pitemize}
\end{definiceN}

\begin{priklad}
Vnitøní a izolované body $X$ nutnì leží v $X$, vnìjší body leží mimo $X$. Hranièní a limitní body $X$ mohou ležet v $X$ i mimo $X$. Vezmìme v euklidovské rovinì $R^2$ množinu $X = \lbrace x \in R^2\;\vert\;0 < \|x\| < 1\rbrace \cup \lbrace(0, 2)\rbrace$, jednotkový kruh se støedem v poèátku, z nìhož jsme poèátek odstranili a k
nìmuž jsme pøidali bod $(0, 2)$. Pak vnitøní body $X$ tvoøí množinu $\lbrace x \in R^2\;\vert\;0 <
\|x\| < 1\rbrace$, vnìjší body množinu $\lbrace x \in R^2\;\vert\; \|x\| > 1, x \neq (0, 2)\rbrace$, hranièní
body množinu $\lbrace x \in R^2\;\vert\; \|x\| = 1\rbrace \cup \lbrace(0, 0), (0, 2)\rbrace$, limitní body množinu
$\lbrace x \in R^2\;\vert\;\|x\| \leq 1\rbrace$ a izolované body množinu $\lbrace(0, 2)\rbrace$.
\end{priklad}

\begin{definiceN}{vzdálenost bodu od množiny}
V metrickém prostoru $(X, d)$ buï $A\subset X$ množina a $x\in(X, d)$ bod. \emph{Vzdálenost bodu $x$ od množiny $A$} je èíslo $ d(x,A)=\inf\{ d(x,y)|y\in A\}$.
\end{definiceN}

\begin{definiceN}{uzávìr}
\emph{Uzávìrem} množiny $A$ v metrickém prostoru $(M,d)$ nazýváme množinu
$$\overline{A} = \bigcap_{\forall F}\{F\;\vert\;F \text{ uzavøená}, A \subseteq F \subseteq M\}$$
Ekvivalentnì $\overline{A} = \lbrace a \in M\;\vert\;a = \lim_{n \to \infty} a_n$ pro nìjakou $(a_n) \subset A \rbrace$, neboli $\overline{A} = A \cup \lbrace \text{limitní body }A\rbrace$.
\end{definiceN}

\begin{vetaN}{vlastnosti uzávìru}
Buï $(M,d)$ metrický prostor a $A,B$ množiny v nìm, potom platí:
\begin{penumerate}
\item $\overline{\emptyset} = \emptyset$, $\overline{M} = M$
\item $A \subset B \Rightarrow \overline{A} \subset \overline{B}$
\item $\overline{\overline{A}} = \overline{A}$
\item $\overline{A \cup B} = \overline{A} \cup \overline{B}$
\item charakteristika uzávìru: $\overline{A} = \{x \in M| d(x,A) = 0\}$
\item $\overline{A}$ je nejmenší uzavøená množina obsahující $A$, proto $A$ je uzavøená právì když $\overline{A}=A$.
\end{penumerate}
\end{vetaN}

\begin{poznamkaN}{ekvivalentní definice uzavøené množiny}
Množina $X\subset(M, d)$ je uzavøená, jestliže každá posloupnost bodù $(x_n)_{n\geq 0}$, která v $X$ leží a konverguje, v $X$ má také svou limitu.
\end{poznamkaN}

\subsubsection*{Spojitost a stejnomìrná spojitost}

\begin{definiceN}{spojité zobrazení}
Pro metrické prostory $(X, d_1)$ a $(Y, d_2)$ je zobrazení $f:X\to Y$ \emph{spojité v bodì} $x\in X$, jestliže
$$\forall\varepsilon>0\ \exists\delta>0:  d_1(x,y)<\delta\;\Rightarrow\;  d_2(f(x),f(y))<\varepsilon$$
Ekvivalentnì pro $\forall$ okolí $V$ bodu $f(x)$ $\exists$ okolí $U$ bodu $x$ tak, že $y \in U \Rightarrow f(y) \in V$.\\
Zobrazení $f$ je \emph{spojité}, pokud je spojité v každém bodì $x\in X$. 
\end{definiceN}

\begin{vetaN}{vlastnosti spojitosti}
Nech je dáno zobrazení $f:X\to Y$ mezi dvìma metrickými prostory $(X, d_1)$, $(Y, d_2)$. Pak jsou následující tvrzení ekvivalentní.
\begin{penumerate}
    \item $f$ je spojité.
    \item Pro každou konvergentní posloupnost $(x_n)_{n\geq 0}$ v $X$ platí $f(\lim x_n)=\lim f(x_n)$.
    \item Pro každé $x$ a každé okolí $U$ bodu $f(x)$ existuje okolí $V$ bodu $x$ takové, že $f[V]\subseteq U$.
    \item Je-li $U \subset Y$ otevøená v $Y$, pak $f^{-1}(U) \subset X$ je otevøená v $X$.
    \item Je-li $U \subset Y$ uzavøená v $Y$, pak $f^{-1}(U) \subset X$ je uzavøená v $X$.
    \item Pro každou $A\subseteq X$ platí $f[\overline{A}]\subseteq \overline{f[A]}$.
\end{penumerate}
\begin{ideadukazu}
\begin{penumerate}
    \item[1$\Rightarrow$2] $f$ je spojité a $(x_n) \to x$ v $X$, chceme ukázat, že pro (každé) dané $\varepsilon > 0$ existuje $n_0$ tak, že $n \geq n_0 \Rightarrow d_2(f(x_n), f(x)) \leq \varepsilon$ (neboli $\lim f(x_n) = f(x) = f(\lim x_n)$). Máme $f$ spojité, proto pro dané $\varepsilon$ existuje $\delta > 0:\;d_1(x_n, x) < \delta \Rightarrow d_2(f(x_n), f(x)) < \varepsilon$. A protože $x_n \to x$ v $X$, máme pro takové $\delta$ vždy nìjaké $n_0:\;n\geq n_0 \Rightarrow d_1(x_n, x) < \delta$. Tím jsme získali hledané $n_0$. (Argumentoval jsem pozpátku od cíle k nalezení potøebného \uv{parametru} $n_0$.)
    
    Ještì jednou -- protože $f$ je spojité, pro libovolnì malou vzdálenost od $f(\lim x_n)$ dostaneme nutnou vzdálenost k~$\lim x_n$, ale k~té už se pøiblížit umíme.
    
    \item[2$\Rightarrow$3] Nepøímo. Nech máme $x$ a $U$ takové, že okolí $V$ neexistuje. Díky tomu vytvoøíme posloupnost, která nesplòuje podmínku 2 (okolí se \uv{nezobrazí} do okolí).
    \item[3$\Rightarrow$4] $U \subset Y$ otevøená, $x \in f^{-1}(U)$, tedy $U$ je okolím $f(x)$, proto existuje okolí $V$ bodu $x$, že $f(V) \subset U$ a tedy $V \subset f^{-1}(U)$ a proto je otevøená.
    \item[4$\Leftrightarrow$5] Vzor doplòku je doplnìk vzoru.
    \item[5$\Rightarrow$6] $A \subseteq f^{-1}\left(f\left(A\right)\right) \subseteq f^{-1}\left(\overline{f\left(A\right)}\right)$ a vzory uzavøených jsou uzavøené, tak máme $\overline{A} \subseteq f^{-1}\left(\overline{f\left(A\right)}\right)$, protože $\overline{A}$ je \textbf{nejmenší} uzavøená množina, obsahující $A$ a $f^{-1}\left(\overline{f\left(A\right)}\right)$ je uzavøená množina, obsahující A. Proto $f\left(\overline{A}\right) \subseteq \overline{f\left(A\right)}$ (ze základní logiky zobrazení).
    \item[6$\Rightarrow$1] Nepøímo. Tedy máme $x, \varepsilon > 0$ tak, že pro všechna $\delta > 0$ existuje $y \in B(x, \delta)$ takové, že $d_2(f(x),f(y)) \geq \varepsilon$. Z nich vytvoøíme $A = \left\lbrace y\;\middle\vert\;d_2\left(f\left(y\right),f\left(x\right)\right) \geq \varepsilon\right\rbrace$. Pak ovšem $x \in \overline{A}$ (protože v~A jsou body $x$ blízko libovolnì malé $\delta$), ale $f(x) \notin  \overline{f(A)}$ (protože od $f(a)$ jsme vždy vzdáleni o $\varepsilon$), což je spor s 6.
\end{penumerate}
\end{ideadukazu}
\end{vetaN}

\begin{definiceN}{stejnomìrnì spojité zobrazení}
Øekneme, že zobrazení $f:(X, d_1)\to(Y, d_2)$ je \emph{stejnomìrnì spojité}, jestliže
$$\forall\varepsilon>0\ \exists\delta>0\text{ takové, že }\forall(x,y)\text{ platí } d_1(x,y)<\delta\;\Rightarrow\; d_2(f(x),f(y))<\varepsilon$$
\end{definiceN}

\begin{vetaN}{skládání zobrazení}
Složení dvou spojitých (nebo stejnomìrnì spojitých) zobrazení je spojité (resp. stejnomìrnì spojité).
\end{vetaN}

\begin{definiceN}{homeomorfismus}
Zobrazení $f:(X,d_1)\to(Y,d_2)$ nazveme \emph{(stejnomìrným) homeomorfismem}, pokud
\begin{pitemize}
    \item $f$ je bijekce,
    \item $f$ je (stejnomìrnì) spojité,
    \item inverzní zobrazení $f^{-1}$ je (stejnomìrnì) spojité.
\end{pitemize}
Prostory $X$ a $Y$ pak jsou \emph{(stejnomìrnì) homeomorfní}. Pokud je $f$ identické zobrazení z $(X, d_1)$ do $(X, d_2)$, øíkáme, že metriky $d_1$ a $ d_2$ jsou \emph{(stejnomìrnì) ekvivalentní}. (Jiná definice ekvivalentních metrik požaduje, aby metrické prostory $(X, d_1)$ a $(X, d_2)$ mìli tytéž otevøené množiny).
\end{definiceN}

\begin{vetaN}{aritmetika zobrazení}
Jsou-li $f,g$ spojité funkce $(X, d)\to\mathbb{R}$ (kde $(X, d)$ je metrický prostor) a $\alpha\in\mathbb{R}$, potom i funkce $f+g$, $\alpha\cdot f$, $f\cdot g$ a $\frac{f}{g}$ (má-li tato smysl) jsou spojité. Platí i pro spojitost zobrazení v nìjakém bodì $x_0\in X$.

\begin{dukaz}
Dùkaz této vìty je vlastnì stejný jako dùkaz vìty o aritmetice limit pro reálné funkce (jen pracujeme se zobrazeními na metrických prostorech).
\end{dukaz}
\end{vetaN}

\subsubsection*{Podprostor metrického prostoru}

\begin{definiceN}{podprostor}
Pro metrický prostor $(X, d)$ a množinu $X_1\subset X$ vezmeme funkci $d_1:X_1\times X_1\to\mathbb{R}$ danou pøedpisem $d_1(x,y)= d(x,y)\ \forall x,y\in X_1$. Pak $(X_1, d_1)$ je \emph{podprostor} metrického prostoru $(X, d)$ (\emph{indukovaný} podmnožinou $X_1$).
\end{definiceN}

\begin{poznamka}
Pro $X_1 \subset X$ je zobrazení vložení $j:(X_1, d)\to(X, d)$, definované pøedpisem $j(x)=x$ stejnomìrnì spojité.
\end{poznamka}

\begin{vetaN}{vlastnosti podprostorù}
Buï $Y$ podprostor metrického prostoru $X$. Potom platí
\begin{penumerate}
    \item $B_Y(x,\varepsilon)=B_X(x,\varepsilon)\cap Y$,\hfill\emph{(o okolí bodù)}
    \item $U$ je otevøená v $Y$, právì když existuje otevøená $V \subset X$ taková, že $U=V\cap Y$ (to samé platí i pro uzavøené množiny),
    \item $\overline{A}^Y=\overline{A}^X\cap Y$.\hfill\emph{(o uzávìru množiny)}
\end{penumerate}
\end{vetaN}

\begin{vetaN}{podprostor zachovává spojitost}
Pro $f:(X, d_1)\to (Y, d_2)$ (stejnomìrnì) spojité zobrazení a $X_1\subseteq X$, $Y_1\subseteq Y$ takové, že $f[X_1]\subseteq Y_1$ je $f_1:X_1\to Y_1$ definované pøedpisem $f_1(x)=f(x)$ (stejnomìrnì) spojité.
\end{vetaN}
