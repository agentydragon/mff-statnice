\subsection{Reálné funkce jedné reálné promìnné}
\begin{e}{Definice}{0}{Reálná funkce}
Reálná funkce jedné promìnné je zobrazení $f: M \rightarrow \mathbb{R}$, kde $M \subseteq \mathbb{R}$. \\
$f$ mùe bıt na $M$:
\begin{pitemize}
        \item \emph{rostoucí}: $\forall x, y: x < y \Rightarrow f(x) < f(y)$
        \item \emph{klesající}: $\forall x, y: x < y \Rightarrow f(x) > f(y)$
        \item \emph{neklesající}: $\forall x, y: x < y \Rightarrow f(x) \le f(y)$
        \item \emph{nerostoucí}: $\forall x, y: x < y \Rightarrow f(x) \ge f(y)$

        \item \emph{sudá}: $x \in M \Rightarrow -x \in M \wedge f(x)=f(-x), \forall x \in M$
        \item \emph{lichá}: $x \in M \Rightarrow -x \in M \wedge f(x)=-f(-x), \forall x \in M$
        \item \emph{periodická} s periodou $p\in\mathbb{R}$: $x \in M \Rightarrow x \pm p \in M \wedge f(x)=f(x+p)=f(x-p), \forall x \in M$
\end{pitemize}
\end{e}

\begin{e}{Definice}{0}{Okolí bodu}
    \begin{pitemize}
        \item $P(a, \delta) = (a-\delta, a) \cup (a, a+\delta)$  \hfill \emph{(prstencové okolí)} \hspace{0.4\textwidth}

        \item $P^{+}(a, \delta)=(a, a+\delta)$  \hfill \emph{(pravé prstencové okolí)} \hspace{0.4\textwidth}

        \item $P^{-}(a, \delta)=(a-\delta, a)$  \hfill \emph{(levé prstencové okolí)} \hspace{0.4\textwidth}
        
        \item $B(a, \delta)=(a-\delta, a+\delta) = P(a, \delta)\cup\{a\}$ \hfill \emph{($\delta$-okolí)} \hspace{0.4\textwidth}

        \item $B^{+}(a, \delta)=P^{+}(a, \delta)\cup\{a\}$ \hfill \emph{(pravé $\delta$-okolí)} \hspace{0.4\textwidth}
        
        \item $B^{-}(a, \delta)=P^{-}(a, \delta)\cup\{a\}$ \hfill \emph{(levé $\delta$-okolí)} \hspace{0.4\textwidth}
        
        \item $P(\infty, d) = B(\infty,d)=(\frac{1}{d}, \infty)$
        \item $P(-\infty, d) = B(-\infty,d)=(-\infty, -\frac{1}{d})$
\end{pitemize}
\end{e}

\begin{e}{Vìta}{1}{Heineho vìta}

\begin{e}{Definice}{0}{0}
$f: M \rightarrow \mathbb{R}, M \subseteq \mathbb{R}$. $f$ nabıvá v bodì $a \in M$:
\begin{pitemize}
        \item \emph{maxima} na M $\Leftrightarrow \forall x \in M: f(x) \le f(a)$
        \item \emph{minima} na M $\Leftrightarrow \forall x \in M: f(x) \ge f(a)$
        \item \emph{ostrého maxima} na M $\Leftrightarrow \forall x \in M\setminus \{a\}: f(x) < f(a)$
        \item \emph{ostrého minima} na M $\Leftrightarrow \forall x \in M\setminus \{a\}: f(x) > f(a)$
        \item \emph{lokálního} maxima (minima, ostrého...)$\Leftrightarrow \exists \delta > 0: f$ nabıvá na $M \bigcap B(a, \delta)$ maxima (minima, ...)
\end{pitemize}
\end{e}

\begin{e}{Definice}{0}{prostá funkce, inverzní funkce}
Funkce $f$ je \emph{prostá}, jestlie $\forall x, y \in D(f): x \neq y \Rightarrow f(x) \neq f(y)$.

Nech $f$ je prostá na $M$, tedy $f: M \rightarrow f(M)$. Pak \emph{inverzní funkce} $f^{-1}$ k funkci $f$ je definovaná na $f(M)$ pøedpisem: $y \in f(M)$, pak $f^{-1}(y)=x \Leftrightarrow y=f(x)$.
\end{e}

\end{e}

\begin{e}{Vìta}{0}{Exponenciální funkce}
Existuje právì jedna reálná funkce \textbf{exp}, splòující:
\begin{pitemize}
        \item $\exp(x+z)=\exp(x) \exp(z), \forall x,z \in \mathbb{R}$
        \item $\exp(x) \ge 1+x, \forall x \in \mathbb{R}$
\end{pitemize}

\begin{e}{Poznámka}{0}{Nìkteré vlastnosti $\exp$}
Platí:
\begin{pitemize}
    \item $\exp 0=1$
    \item $\exp (-x) = \frac{1}{\exp x}$
    \item $\exp (x) \neq 0\ \forall x\in \mathbb{R}$
    \item $\lim_{x\to\infty}\exp x=\infty$, $\lim_{x\to-\infty}\exp x=0$
    \item $\exp$ je rostoucí na $\mathbb{R}$
    \item $\lim_{x\to 0}\frac{\exp x - 1}{x}=1$
\end{pitemize}
\end{e}

\begin{e}{Vìta}{0}{Vlastnosti $\log$}

\begin{pitemize}

        \item $\log(x\cdot y) = \log x + \log y, \forall x, y \in (0, \infty)$, $\log(x^n)=n\log x$
        \item log je spojitá, rostoucí na $(0, \infty)$, $\log 1=0$, $\log e = 1$
        \item $lim_{x \rightarrow 0_{+}} \log x = - \infty$, $lim_{x \rightarrow \infty} \log x = \infty$
        \item $lim_{x \rightarrow 1} \frac{\log x}{x-1} = 1$, $lim_{x \rightarrow 0} \frac{\log x+1}{x} = 1$
\end{pitemize}
\end{e}

\begin{e}{Vìta}{0}{O derivaci sloené funkce}

\begin{e}{Vìta}{0}{O derivaci inverzní funkce}
Nech funkce $f$ je na intervalu $(a,b)$ spojitá a ryze monotonní\footnote{aby vùbec mìla inverzní}  a má v bodì $x_0 \in (a,b)$ derivaci $f'(x_0)$ vlastní a rùznou od nuly. Potom má funkce $f^{-1}$ derivaci v bodì $y_0=f(x_0)$ a platí rovnost
$$(f^{-1})'(y_0) = \frac{1}{f'(f^{-1}(y_0))}=\frac{1}{f'(x_0)}.$$
\begin{e}{Poznámka}{0}{0}
Pokud v situaci popsané v právì uvedené vìtì je $f'(x_0)$ nevlastní, je $(f^{-1})'(y_0)=0$. Je-li $f'(x_0)=0$, je $(f^{-1})'(y_0)=+\infty$ (je-li $f$ rostoucí), resp. $=-\infty$ (je-li $f$ klesající).
\end{e}

\begin{e}{Vìta}{0}{L'Hospitalovo pravidlo}

\begin{e}{Vìta}{0}{Vztah derivace a monotonie}

        \item je-li $f' < 0$ na $Int(I)$, pak f je klesající na I

\end{penumerate}
\end{e}

\begin{e}{Definice}{0}{Teèna, inflexe}
Nech funkce $f$ má v $a \in \mathbb{R}$ vlastní derivaci. Oznaèíme $T_a=\{[x,y]\in \mathbb{R}^2, y=f(a)+f'(a)(x-a)\}$. Øekneme, e $[x, f(x)] \in \mathbb{R}^2$ leí nad (pod) teènou $T_a$, jestlie $f(x)>f(a)+f'(a)(x-a)$ ($f(x)<f(a)+f'(a)(x-a)$).

\end{e}

\end{e}

\end{e}

\subsubsection*{Taylorùv polynom se zbytkem}

\begin{e}{Definice}{0}{0}
Nech $f$ je reálná funkce, $a \in \mathbb{R}$, $n \in \mathbb{N}$ a $f$ má derivace do øádu $n$. Pak funkci
$$T^{f,a}_n(x)=f(a) + f'(a)(x-a) + \dots + \frac{f^{(n)}(a)}{n!}(x-a)^n$$
nazıváme \emph{Taylorovım polynomem funkce $f$ øádu $n$ v bodì $a$}.
\end{e}

\begin{e}{Vìta}{0}{0}
Nech $f$ je reálná funkce, $a \in \mathbb{R}$, nech existuje vlastní $f^{(n)}(a)$. Nech $P$ je polynom stupnì $\le n$. Pak $$\lim_{x \rightarrow a} \frac{f(x)-P(x)}{(x-a)^n}=0 \Leftrightarrow P=T^{f,a}_{n}$$
\end{e}

\begin{e}{Vìta}{0}{Obecnı tvar zbytku}
Nech $f$ má vlastní (n+1)-ní derivaci v intervalu $\left<a,x\right>, x>a$. Nech $\varphi$ je spojitá funkce na $\left<a,x\right>$, která má na $(a,x)$ vlastní nenulové derivace. Pak:
$$\exists \xi \in (a, x): R^{f,a}_{n}(x)=f(x)-T^{f,a}_n(x)=\frac{1}{n!}.\frac{\varphi(x)-\varphi(a)}{\varphi'(\xi)}.f^{(n+1)}(\xi).(x-\xi)^n$$

\begin{e}{Dùkaz}{0}{0}
Vìta je dùsledkem Cauchyho vìty o støední hodnotì, aplikované na funkci $F(t):=f(x)-T^{f,t}_n(x)$, definované pro $t\in[a,x]$. (Ošklivou a pracnou) derivací této funkce vyjde, e $F'(t) = -\frac{f^{(n+1)}(t)}{n!}(x-t)^n\ \forall t\in(a,x)$ a teï pouijeme onu Cauchyho vìtu a dostaneme 
$$\frac{-\frac{f^{(n+1)}(\xi)}{n!}(x-\xi)^n}{\varphi'(\xi)} = \frac{F'(\xi)}{\varphi'(\xi)} = \frac{F(x)-F(a)}{\varphi(x)-\varphi(a)} = \frac{0-R_n^{f,a}(x)}{\varphi(x)-\varphi(a)}$$
co u dává kıenı tvar zbytku.
\begin{flushright}$\square$\end{flushright}
\end{e}
\end{e}

\begin{e}{Dùsledek}{0}{0}
\emph{Lagrangeùv tvar zbytku}:
Je-li $\varphi(t)=(x-t)^{n+1}$, dostaneme
$$R^{f,a}_{n}(x)=\frac{1}{(n+1)!} f^{(n+1)}(\xi) (x-a)^{n+1}$$
\emph{Cauchyho tvar zbytku}:
Je-li $\varphi(t)=t$, dostaneme
$$R^{f,a}_{n}(x)=\frac{1}{n!} f^{(n+1)}(\xi) (x-\xi)^{n} (x-a)$$
\end{e}
