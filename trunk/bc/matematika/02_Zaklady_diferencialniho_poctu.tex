\subsection{Reálné funkce jedné reálné promìnné}
\begin{definiceN}{Reálná funkce}
Reálná funkce jedné promìnné je zobrazení $f: M \rightarrow \mathbb{R}$, kde $M \subseteq \mathbb{R}$. \\
$f$ je na $M$:
\begin{pitemize}
	\item \emph{rostoucí}: $\forall x, y: x < y \Rightarrow f(x) < f(y)$
	\item \emph{klesající}: $\forall x, y: x < y \Rightarrow f(x) > f(y)$
	\item \emph{neklesající}: $\forall x, y: x < y \Rightarrow f(x) \le f(y)$
	\item \emph{nerostoucí}: $\forall x, y: x < y \Rightarrow f(x) \ge f(y)$

	\item \emph{sudá}: $x \in M \Rightarrow -x \in M \wedge f(x)=f(-x), \forall x \in M$
	\item \emph{lichá}: $x \in M \Rightarrow -x \in M \wedge f(x)=-f(-x), \forall x \in M$
	\item \emph{periodická} s periodou $p\in\mathbb{R}$: $x \in M \Rightarrow x \pm p \in M \wedge f(x)=f(x+p)=f(x-p), \forall x \in M$
\end{pitemize}
\end{definiceN}

\begin{definiceN}{Okolí bodu}
    \begin{pitemize}
        \item $P(a, \delta) = (a-\delta, a) \cup (a, a+\delta)$  \hfill \emph{(prstencové okolí)} \hspace{0.4\textwidth}

        \item $P^{+}(a, \delta)=(a, a+\delta)$  \hfill \emph{(pravé prstencové okolí)} \hspace{0.4\textwidth}

        \item $P^{-}(a, \delta)=(a-\delta, a)$  \hfill \emph{(levé prstencové okolí)} \hspace{0.4\textwidth}
        
        \item $B(a, \delta)=(a-\delta, a+\delta) = P(a, \delta)\cup\{a\}$ \hfill \emph{($\delta$-okolí)} \hspace{0.4\textwidth}

        \item $B^{+}(a, \delta)=P^{+}(a, \delta)\cup\{a\}$ \hfill \emph{(pravé $\delta$-okolí)} \hspace{0.4\textwidth}
        
        \item $B^{-}(a, \delta)=P^{-}(a, \delta)\cup\{a\}$ \hfill \emph{(levé $\delta$-okolí)} \hspace{0.4\textwidth}
        
        \item $P(\infty, d) = B(\infty,d)=(\frac{1}{d}, \infty)$
        \item $P(-\infty, d) = B(-\infty,d)=(-\infty, -\frac{1}{d})$
\end{pitemize}
\end{definiceN}

\begin{vetaN}{Heineho vìta}
Nech  $f: M \rightarrow \mathbb{R}, M \subseteq \mathbb{R}$. Nech $f$ je definováno na nìjakém prstencovém okolí bodu $a \in \mathbb{R}^{*}$. Potom následující dvì tvrzení jsou ekvivalentní:
\begin{penumerate}
	\item $lim_{x \rightarrow a} f(x) = A \in \mathbb{R}^{*}$
	\item Pro kadou posloupnost $\{x_n\}_{n=1}^{\infty}$ splòující $x_n \in D(f)\ \forall n \in \mathbb{N}$ a  $\lim x_n = a, x_n \neq a\ \forall n$ platí $\lim_{n \rightarrow \infty} f(x_n)=A$
\end{penumerate}
Heineho vìta umoòuje tvrzení, vyslovená o limitách posloupností, pøevádìt na limity funkcí v bodì.

\begin{ideadukazu}
    (nejsem si jist, jestli dùkaz funguje i pro nekoneèna)

    2$\Rightarrow$1 Sporem. Musí existovat $\varepsilon$, e $\forall \delta \exists x \in P(a, \delta)$ takové, e $f(x) \notin B(A, \varepsilon)$ (pouze negace definice limity). Mùeme tedy postupnì zmenšovat $\delta$, v~kadém $ P(a, \delta)$ najít takové $x$ a vytvoøit tak posloupnost, která má limitu v~$a$, ale její obrazy nebudou mít limitu v~$A$, protoe jsou vdy vzdálené $\varepsilon$ -- spor.
\end{ideadukazu}
\end{vetaN}

\begin{definiceN}{Jednostranná spojitost funkce v bodì}
\begin{pitemize}
	\item $\textit{funkce f je spojitá v a} \Leftrightarrow \lim_{x \rightarrow a} f(x)=f(a) \Leftrightarrow \forall \varepsilon>0~\exists \delta >0: f(B(a, \delta)) \subseteq B(f(a), \varepsilon)$
	\item $\textit{funkce f je spojitá v a zprava} \Leftrightarrow \lim_{x \rightarrow a^{+}} f(x)=f(a) \Leftrightarrow \forall \varepsilon>0~\exists \delta >0: f(B^{+}(a, \delta)) \subseteq B(f(a), \varepsilon)$
	\item $\textit{funkce f je spojitá v a zleva} \Leftrightarrow \lim_{x \rightarrow a^{-}} f(x)=f(a) \Leftrightarrow \forall \varepsilon>0~\exists \delta >0: f(B^{-}(a, \delta)) \subseteq B(f(a), \varepsilon)$
\end{pitemize}
\end{definiceN}

\begin{vetaN}{O limitì sloené funkce}
Nech $\lim_{x \rightarrow a} g(x) = A$, $\lim_{y \rightarrow A} f(y) = B$, $a, A, B \in \mathbb{R}^{*}$. Nech navíc platí jeden z pøedpokladù:\\
(P1) $f$ je spojitá v $A$\hfill \textit{(vnìjší funkce je spojitá)}\\
(P2) $\exists \delta > 0: g(x) \neq A$ pro $\forall x \in P(a, \delta)$\hfill \textit{(vnitøní je \uv{lokálnì prostá})}\\
Potom $\lim_{x \rightarrow a} (f \circ g)(x) = B$.
\end{vetaN}

\begin{definiceN}{Interval}
Nech $a, b \in \mathbb{R}^{*}, a < b$. Pak \emph{otevøenım intervalem} $(a, b)$ nazveme $\{x | a<x<b\}$, \emph{(uzavøenım) intervalem} $\left<a, b\right>$ (pro $a, b \in \mathbb{R}$) nazveme $I=\{x | a \le x \le b\}$. Uzavøenı interval se nìkdy znaèí i $[a,b]$.
\end{definiceN}

\begin{vetaN}{Vìta o limitì monotónní funkce}
Nech je funkce $f$ monotónní na $(a, b)$, $a, b \in \mathbb{R}^{*}, a < b$. Potom $\exists \lim_{x \rightarrow a^{+}} f(x)$, $\exists \lim_{x \rightarrow b^{-}} f(x)$.
\end{vetaN}

\begin{definiceN}{Spojitost na intervalu}
Je-li $\left<a, b\right>$ interval, pak $a$ nazıváme poèáteèním bodem, $b$ koncovım bodem a $x \in (a, b)$ vnitøními body.

Øekneme, e $f$ \emph{je spojitá na intervalu $I$}, jestlie je spojitá zprava ve všech bodech kromì koncového a spojitá zleva ve všech bodech kromì poèáteèního.
\end{definiceN}

\begin{vetaN}{O spojitém obrazu intervalu}
Nech funkce $f: I \rightarrow \mathbb{R}$ je spojitá na intervalu $I$. Potom $f(I)$ je také interval.
\end{vetaN}

\begin{vetaN}{Darbouxova o nabıvaní mezihodnot}
Nech funkce $f$ je spojitá na $\left<a, b\right>$ a $f(a)<f(b)$, potom $\forall y \in (f(a), f(b))$ existuje $x \in (a, b)$ takové, e $f(x)=y$.
\end{vetaN}

\begin{definice}
$f: M \rightarrow \mathbb{R}, M \subseteq \mathbb{R}$. $f$ nabıvá v bodì $a \in M$:
\begin{pitemize}
	\item \emph{maxima} na M $\Leftrightarrow \forall x \in M: f(x) \le f(a)$
	\item \emph{minima} na M $\Leftrightarrow \forall x \in M: f(x) \ge f(a)$
	\item \emph{ostrého maxima} na M $\Leftrightarrow \forall x \in M\setminus \{a\}: f(x) < f(a)$
	\item \emph{ostrého minima} na M $\Leftrightarrow \forall x \in M\setminus \{a\}: f(x) > f(a)$
	\item \emph{lokálního} maxima (minima, ostrého...)$\Leftrightarrow \exists \delta > 0: f$ nabıvá na $M \bigcap B(a, \delta)$ maxima (minima, ...)
\end{pitemize}
\end{definice}

\begin{vetaN}{Vztah spojitosti a extrémù}
Nech $f$ je spojitá na $\left<a, b\right>$. Potom $f$ nabıvá na $\left<a, b\right>$ svého maxima i minima.
\end{vetaN}

\begin{vetaN}{Vztah spojitosti a omezenosti}
Spojitá funkce na uzavøeném intervalu $\left<a, b\right>$ je omezená.
\end{vetaN}

\begin{definiceN}{prostá funkce, inverzní funkce}
Funkce $f$ je \emph{prostá}, jestlie $\forall x, y \in D(f): x \neq y \Rightarrow f(x) \neq f(y)$.

Nech $f$ je prostá na $M$, tedy $f: M \rightarrow f(M)$. Pak \emph{inverzní funkce} $f^{-1}$ k funkci $f$ je definovaná na $f(M)$ pøedpisem: $y \in f(M)$, pak $f^{-1}(y)=x \Leftrightarrow y=f(x)$.
\end{definiceN}

\begin{vetaN}{O inverzní funkci}
Nech $I$ je interval a funkce $f$ je definovaná, spojitá a rostoucí (klesající) na $I$. Potom inverzní funkce $f^{-1}$ je definována, spojitá a rostoucí (klesající) na $f(I)$.
\end{vetaN}

\begin{vetaN}{Exponenciální funkce}
Existuje právì jedna reálná funkce \textbf{exp}, splòující:
\begin{pitemize}
	\item $\exp(x+z)=\exp(x) \exp(z), \forall x,z \in \mathbb{R}$
	\item $\exp(x) \ge 1+x, \forall x \in \mathbb{R}$
\end{pitemize}
\end{vetaN}

\begin{poznamkaN}{Nìkteré vlastnosti $\exp$}
Platí:
\begin{pitemize}
    \item $\exp 0=1$
    \item $\exp (-x) = \frac{1}{\exp x}$
    \item $\exp (x) \neq 0\ \forall x\in \mathbb{R}$
    \item $\lim_{x\to\infty}\exp x=\infty$, $\lim_{x\to-\infty}\exp x=0$
    \item $\exp$ je rostoucí na $\mathbb{R}$
    \item $\lim_{x\to 0}\frac{\exp x - 1}{x}=1$
\end{pitemize}
\end{poznamkaN}

\begin{vetaN}{Vlastnosti $\log$}
Funkce $\log$, definovaná pøedpisem $\log=\exp^{-1}$ má následující vlastnosti:
\begin{pitemize}
	\item $D(\log)=(0, \infty)$, $\log ((0, \infty)) \rightarrow \mathbb{R}$
	\item $\log(x\cdot y) = \log x + \log y, \forall x, y \in (0, \infty)$, $\log(x^n)=n\log x$
	\item log je spojitá, rostoucí na $(0, \infty)$, $\log 1=0$, $\log e = 1$
	\item $lim_{x \rightarrow 0_{+}} \log x = - \infty$, $lim_{x \rightarrow \infty} \log x = \infty$
	\item $lim_{x \rightarrow 1} \frac{\log x}{x-1} = 1$, $lim_{x \rightarrow 0} \frac{\log x+1}{x} = 1$
\end{pitemize}
\end{vetaN}

\begin{definice}

\begin{pitemize}

	\item $\arccos x = y \Leftrightarrow y \in <0, \pi> \wedge \cos y=x$

\end{definice}

\subsection{Derivace: definice a základní pravidla, vìty o støední hodnotì, derivace vyšších øádù}
\begin{definice}
Nech $f$ je reálná funkce jedné promìnné, $a \in \mathbb{R}$. Derivací funkce $f$ v bodì $a$ nazveme
$$f'(a) = \lim_{h \rightarrow 0} \frac{f(a+h)-f(a)}{h} \textrm{, pokud existuje}$$

\noindent Derivací zprava a zleva rozumíme:
$$f'_{+}(a) = \lim_{h \rightarrow 0_+} \frac{f(a+h)-f(a)}{h}, \, f'_{-}(a) = \lim_{h \rightarrow 0_-} \frac{f(a+h)-f(a)}{h}$$
\end{definice}

\begin{vetaN}{Vztah derivace a spojitosti}
Má-li $f$ v $a$ vlastní derivaci, potom je $f$ v $a$ spojitá.
\end{vetaN}

\begin{vetaN}{Aritmetika derivací}
Nech existují $f'(a)$, $g'(a)$:
\begin{penumerate}
	\item $(f+g)'(a)=f'(a)+g'(a)$, je-li pravá strana definována
	\item je-li $g$ nebo $f$ spojitá v $a$, pak $(fg)'(a)=f'(a)g(a)+f(a)g'(a)$
	\item je-li $g$ spojitá v $a$, $g(a) \neq 0$, pak $(\frac{f}{g})'(a)=\frac{f'(a)g(a)-f(a)g'(a)}{g^2(a)}$
\end{penumerate}
\end{vetaN}

\begin{vetaN}{O derivaci sloené funkce}
Nech funkce $f$ má derivaci v $y_0$, $g$ má derivaci v $x_0$, $g$ je spojitá v $x_0$ a $y_0=g(x_0)$. Potom $(f \circ g)'(x_0)=f'(y_0)\cdot g'(x_0)=f'(g(x_0))\cdot g'(x_0)$.
\end{vetaN}

\begin{vetaN}{O derivaci inverzní funkce}
Nech funkce $f$ je na intervalu $(a,b)$ spojitá a ryze monotonní a má v bodì $x_0 \in (a,b)$ derivaci $f'(x_0)$ vlastní a rùznou od nuly. Potom má funkce $f^{-1}$ derivaci v bodì $y_0=f(x_0)$ a platí rovnost
$$(f^{-1})'(y_0) = \frac{1}{f'(f^{-1}(y_0))}.$$
\begin{poznamka}
Pokud v situaci popsané v právì uvedené vìtì je $f'(x_0)$ nevlastní, je $(f^{-1})'(y_0)=0$. Je-li $f'(x_0)=0$, je $(f^{-1})'(y_0)=+\infty$ (je-li $f$ rostoucí), resp. $=-\infty$ (je-li $f$ klesající).
\end{poznamka}
\end{vetaN}

\begin{vetaN}{Nutná podmínka lokálního extrému}
Nech $M \subseteq \mathbb{R}$, $f: M \rightarrow \mathbb{R}$. Nech $f$ má v $a \in M$ lokální extrém. Pak buï neexistuje $f'(a)$, nebo $f'(a)=0$.
\end{vetaN}

\begin{vetaN}{Rolleova vìta}
Nech $f$ je spojitá na $\left<a, b\right>$ a nech existuje $f'(x)\ \forall x \in (a,b)$. Nech $f(a)=f(b)$. Potom existuje $\xi \in (a,b): f'(\xi)=0$.
\end{vetaN}

\begin{vetaN}{Lagrangeova o støední hodnotì}
Nech $f$ je spojitá na $\left<a, b\right>$ a nech existuje $f'(x)\ \forall x \in (a,b)$. Potom existuje $$\xi \in (a,b): f'(\xi)=\frac{f(b)-f(a)}{b-a}$$
\end{vetaN}

\begin{vetaN}{Cauchyova o støední hodnotì}
Nech $f$ a $g$ jsou spojité na $\left<a, b\right>$, nech existuje $f'(x)\ \forall x \in (a,b)$ a nech existuje $g'(x)$ vlastní a nenulové. Potom existuje $$\xi \in (a,b): \frac{f(b)-f(a)}{g(b)-g(a)}=\frac{f'(\xi)}{g'(\xi)}$$

\begin{dukaz}
$g(a) \neq g(b)$, nebo jinak by podle Rolleovy vìty existovalo $\xi \in (a,b): g'(\xi)=0$. Definujeme $H(x)=(f(x)-f(a))(g(b)-g(a))-(g(x)-g(a))(f(b)-f(a))$. Potom $H(a)=H(b)=0$, $H$ je spojitá na $\left<a, b\right> \Rightarrow \exists H' na (a,b)$. Tedy podle Rolleovy vìty $\exists \xi: 0 = H'(\xi)=f'(\xi)(g(b)-g(a))-g'(\xi)(f(b)-f(a))$. Tedy $\frac{f(b)-f(a)}{g(b)-g(a)}=\frac{f'(\xi)}{g'(\xi)}$, nebo $g'(\xi) \neq 0$ a $g(b)-g(a)\neq 0$ 
\begin{flushright}$\square$\end{flushright}
\end{dukaz}
\end{vetaN}

\begin{definice}

\end{definice}

\begin{veta}

\begin{vetaN}{Obecnı tvar zbytku}
Nech $f$ má vlastní (n+1)-ní derivaci v intervalu $\left<a,x\right>, x>a$. Nech $\varphi$ je spojitá funkce na $\left<a,x\right>$, která má na $(a,x)$ vlastní nenulové derivace. Pak:
$$\exists \xi \in (a, x): R^{f,a}_{n}(x)=f(x)-T^{f,a}_n(x)=\frac{1}{n!}.\frac{\varphi(x)-\varphi(a)}{\varphi'(\xi)}.f^{(n+1)}(\xi).(x-\xi)^n$$

\begin{dukaz}
Vìta je dùsledkem Cauchyho vìty o støední hodnotì, aplikované na funkci $F(t):=f(x)-T^{f,t}_n(x)$, definované pro $t\in[a,x]$. (Ošklivou a pracnou) derivací této funkce vyjde, e $F'(t) = -\frac{f^{(n+1)}(t)}{n!}(x-t)^n\ \forall t\in(a,x)$ a teï pouijeme onu Cauchyho vìtu a dostaneme 
$$\frac{-\frac{f^{(n+1)}(\xi)}{n!}(x-\xi)^n}{\varphi'(\xi)} = \frac{F'(\xi)}{\varphi'(\xi)} = \frac{F(x)-F(a)}{\varphi(x)-\varphi(a)} = \frac{0-R_n^{f,a}(x)}{\varphi(x)-\varphi(a)}$$
co u dává kıenı tvar zbytku.
\begin{flushright}$\square$\end{flushright}
\end{dukaz}
\end{vetaN}

Je-li $\varphi(t)=t$, dostaneme

\subsection{Reálné funkce jedné reálné promìnné}
\begin{definiceN}{Reálná funkce}
Reálná funkce jedné promìnné je zobrazení $f: M \rightarrow \mathbb{R}$, kde $M \subseteq \mathbb{R}$. \\
$f$ je na $M$:
\begin{pitemize}
        \item \emph{rostoucí}: $\forall x, y: x < y \Rightarrow f(x) < f(y)$
        \item \emph{klesající}: $\forall x, y: x < y \Rightarrow f(x) > f(y)$
        \item \emph{neklesající}: $\forall x, y: x < y \Rightarrow f(x) \le f(y)$
        \item \emph{nerostoucí}: $\forall x, y: x < y \Rightarrow f(x) \ge f(y)$

        \item \emph{sudá}: $x \in M \Rightarrow -x \in M \wedge f(x)=f(-x), \forall x \in M$
        \item \emph{lichá}: $x \in M \Rightarrow -x \in M \wedge f(x)=-f(-x), \forall x \in M$
        \item \emph{periodická} s periodou $p\in\mathbb{R}$: $x \in M \Rightarrow x \pm p \in M \wedge f(x)=f(x+p)=f(x-p), \forall x \in M$
\end{pitemize}
\end{definiceN}

\begin{definiceN}{Okolí bodu}
    \begin{pitemize}
        \item $P(a, \delta) = (a-\delta, a) \cup (a, a+\delta)$  \hfill \emph{(prstencové okolí)} \hspace{0.4\textwidth}

        \item $P^{+}(a, \delta)=(a, a+\delta)$  \hfill \emph{(pravé prstencové okolí)} \hspace{0.4\textwidth}

        \item $P^{-}(a, \delta)=(a-\delta, a)$  \hfill \emph{(levé prstencové okolí)} \hspace{0.4\textwidth}
        
        \item $B(a, \delta)=(a-\delta, a+\delta) = P(a, \delta)\cup\{a\}$ \hfill \emph{($\delta$-okolí)} \hspace{0.4\textwidth}

        \item $B^{+}(a, \delta)=P^{+}(a, \delta)\cup\{a\}$ \hfill \emph{(pravé $\delta$-okolí)} \hspace{0.4\textwidth}
        
        \item $B^{-}(a, \delta)=P^{-}(a, \delta)\cup\{a\}$ \hfill \emph{(levé $\delta$-okolí)} \hspace{0.4\textwidth}
        
        \item $P(\infty, d) = B(\infty,d)=(\frac{1}{d}, \infty)$

\end{pitemize}
\end{definiceN}

        \item $\lim (f(x)g(x)) = A\cdot B$, je-li vıraz na pravé stranì definován.
        \item $\lim \frac{f(x)}{g(x)} = \frac{A}{B}$, je-li vıraz na pravé stranì definován.
\end{penumerate}
\end{vetaN}

\begin{vetaN}{Limita a uspoøádání - policajti pro funkce}
\begin{penumerate}
        \item Nech $\lim_{x \rightarrow a} f(x) > \lim_{x \rightarrow a} g(x)$, $a \in R^{*}$.\\Potom $\exists P(a, \delta): f(x)>g(x) \ \forall x \in P(a, \delta)$
        \item Nech $f(x) \le g(x), \forall x \in P(a, \delta), \delta > 0$ a existují $\lim_{x \rightarrow a} f(x)$, $\lim_{x \rightarrow a} g(x)$.\\Potom $\lim_{x\to a} f(x) \le \lim_{x\to a} g(x)$.

\end{penumerate}
Pozor na ostrost nerovností, v tomto pøípadì je velmi dùleitá.
\end{vetaN}

\begin{definiceN}{Jednostranná spojitost funkce v bodì}
\begin{pitemize}
        \item $\textit{funkce f je spojitá v a} \Leftrightarrow \lim_{x \rightarrow a} f(x)=f(a) \Leftrightarrow \forall \varepsilon>0~\exists \delta >0: f(B(a, \delta)) \subseteq B(f(a), \varepsilon)$
        \item $\textit{funkce f je spojitá v a zprava} \Leftrightarrow \lim_{x \rightarrow a^{+}} f(x)=f(a) \Leftrightarrow \forall \varepsilon>0~\exists \delta >0: f(B^{+}(a, \delta)) \subseteq B(f(a), \varepsilon)$
        \item $\textit{funkce f je spojitá v a zleva} \Leftrightarrow \lim_{x \rightarrow a^{-}} f(x)=f(a) \Leftrightarrow \forall \varepsilon>0~\exists \delta >0: f(B^{-}(a, \delta)) \subseteq B(f(a), \varepsilon)$
\end{pitemize}
\end{definiceN}

\begin{vetaN}{O limitì sloené funkce}
Nech $\lim_{x \rightarrow a} g(x) = A$, $\lim_{y \rightarrow A} f(y) = B$, $a, A, B \in \mathbb{R}^{*}$. Nech navíc platí jeden z pøedpokladù:\\
(P1) $f$ je spojitá v $A$\hfill \textit{(vnìjší funkce je spojitá)}\\
(P2) $\exists \delta > 0: g(x) \neq A$ pro $\forall x \in P(a, \delta)$\hfill \textit{(vnitøní je \uv{lokálnì prostá})}\\
Potom $\lim_{x \rightarrow a} (f \circ g)(x) = B$.
\end{vetaN}

\begin{definiceN}{Interval}
Nech $a, b \in \mathbb{R}^{*}, a < b$. Pak \emph{otevøenım intervalem} $(a, b)$ nazveme $\{x | a<x<b\}$, \emph{(uzavøenım) intervalem} $\left<a, b\right>$ (pro $a, b \in \mathbb{R}$) nazveme $I=\{x | a \le x \le b\}$. Uzavøenı interval se nìkdy znaèí i $[a,b]$.
\end{definiceN}

\begin{vetaN}{Vìta o limitì monotónní funkce}
Nech je funkce $f$ monotónní na $(a, b)$, $a, b \in \mathbb{R}^{*}, a < b$. Potom $\exists \lim_{x \rightarrow a^{+}} f(x)$, $\exists \lim_{x \rightarrow b^{-}} f(x)$.
\end{vetaN}

\begin{definiceN}{Spojitost na intervalu}
Je-li $\left<a, b\right>$ interval, pak $a$ nazıváme poèáteèním bodem, $b$ koncovım bodem a $x \in (a, b)$ vnitøními body.

Øekneme, e $f$ \emph{je spojitá na intervalu $I$}, jestlie je spojitá zprava ve všech bodech kromì koncového a spojitá zleva ve všech bodech kromì poèáteèního.
\end{definiceN}

\begin{vetaN}{O spojitém obrazu intervalu}
Nech funkce $f: I \rightarrow \mathbb{R}$ je spojitá na intervalu $I$. Potom $f(I)$ je také interval.
\end{vetaN}

\begin{vetaN}{Darbouxova o nabıvaní mezihodnot}
Nech funkce $f$ je spojitá na $\left<a, b\right>$ a $f(a)<f(b)$, potom $\forall y \in (f(a), f(b))$ existuje $x \in (a, b)$ takové, e $f(x)=y$.
\end{vetaN}

\begin{definice}
$f: M \rightarrow \mathbb{R}, M \subseteq \mathbb{R}$. $f$ nabıvá v bodì $a \in M$:
\begin{pitemize}
        \item \emph{maxima} na M $\Leftrightarrow \forall x \in M: f(x) \le f(a)$
        \item \emph{minima} na M $\Leftrightarrow \forall x \in M: f(x) \ge f(a)$
        \item \emph{ostrého maxima} na M $\Leftrightarrow \forall x \in M\setminus \{a\}: f(x) < f(a)$
        \item \emph{ostrého minima} na M $\Leftrightarrow \forall x \in M\setminus \{a\}: f(x) > f(a)$
        \item \emph{lokálního} maxima (minima, ostrého...)$\Leftrightarrow \exists \delta > 0: f$ nabıvá na $M \bigcap B(a, \delta)$ maxima (minima, ...)
\end{pitemize}
\end{definice}

\begin{vetaN}{Vztah spojitosti a extrémù}
Nech $f$ je spojitá na $\left<a, b\right>$. Potom $f$ nabıvá na $\left<a, b\right>$ svého maxima i minima.
\end{vetaN}

\begin{vetaN}{Vztah spojitosti a omezenosti}
Spojitá funkce na uzavøeném intervalu $\left<a, b\right>$ je omezená.
\end{vetaN}

\begin{definiceN}{prostá funkce, inverzní funkce}
Funkce $f$ je \emph{prostá}, jestlie $\forall x, y \in D(f): x \neq y \Rightarrow f(x) \neq f(y)$.

Nech $f$ je prostá na $M$, tedy $f: M \rightarrow f(M)$. Pak \emph{inverzní funkce} $f^{-1}$ k funkci $f$ je definovaná na $f(M)$ pøedpisem: $y \in f(M)$, pak $f^{-1}(y)=x \Leftrightarrow y=f(x)$.
\end{definiceN}

\begin{vetaN}{O inverzní funkci}
Nech $I$ je interval a funkce $f$ je definovaná, spojitá a rostoucí (klesající) na $I$. Potom inverzní funkce $f^{-1}$ je definována, spojitá a rostoucí (klesající) na $f(I)$.
\end{vetaN}

\subsection{Derivace: definice a základní pravidla, vìty o støední hodnotì, derivace vyšších øádù}
\begin{definice}
Nech $f$ je reálná funkce jedné promìnné, $a \in \mathbb{R}$. Derivací funkce $f$ v bodì $a$ nazveme
$$f'(a) = \lim_{h \rightarrow 0} \frac{f(a+h)-f(a)}{h} \textrm{, pokud existuje}$$

\noindent Derivací zprava a zleva rozumíme:
$$f'_{+}(a) = \lim_{h \rightarrow 0_+} \frac{f(a+h)-f(a)}{h}, \, f'_{-}(a) = \lim_{h \rightarrow 0_-} \frac{f(a+h)-f(a)}{h}$$
\end{definice}

\begin{vetaN}{Vztah derivace a spojitosti}
Má-li $f$ v $a$ vlastní derivaci, potom je $f$ v $a$ spojitá.
\end{vetaN}

\begin{vetaN}{Aritmetika derivací}
Nech existují $f'(a)$, $g'(a)$:
\begin{penumerate}
        \item $(f+g)'(a)=f'(a)+g'(a)$, je-li pravá strana definována
        \item je-li $g$ nebo $f$ spojitá v $a$, pak $(fg)'(a)=f'(a)g(a)+f(a)g'(a)$
        \item je-li $g$ spojitá v $a$, $g(a) \neq 0$, pak $(\frac{f}{g})'(a)=\frac{f'(a)g(a)-f(a)g'(a)}{g^2(a)}$
\end{penumerate}
\end{vetaN}

\begin{vetaN}{O derivaci sloené funkce}
Nech funkce $f$ má derivaci v $y_0$, $g$ má derivaci v $x_0$, $g$ je spojitá v $x_0$ a $y_0=g(x_0)$. Potom $(f \circ g)'(x_0)=f'(y_0)\cdot g'(x_0)=f'(g(x_0))\cdot g'(x_0)$.
\end{vetaN}

\begin{vetaN}{O derivaci inverzní funkce}
Nech funkce $f$ je na intervalu $(a,b)$ spojitá a ryze monotonní a má v bodì $x_0 \in (a,b)$ derivaci $f'(x_0)$ vlastní a rùznou od nuly. Potom má funkce $f^{-1}$ derivaci v bodì $y_0=f(x_0)$ a platí rovnost
$$(f^{-1})'(y_0) = \frac{1}{f'(f^{-1}(y_0))}.$$
\begin{poznamka}
Pokud v situaci popsané v právì uvedené vìtì je $f'(x_0)$ nevlastní, je $(f^{-1})'(y_0)=0$. Je-li $f'(x_0)=0$, je $(f^{-1})'(y_0)=+\infty$ (je-li $f$ rostoucí), resp. $=-\infty$ (je-li $f$ klesající).
\end{poznamka}
\end{vetaN}

\begin{vetaN}{Nutná podmínka lokálního extrému}
Nech $M \subseteq \mathbb{R}$, $f: M \rightarrow \mathbb{R}$. Nech $f$ má v $a \in M$ lokální extrém. Pak buï neexistuje $f'(a)$, nebo $f'(a)=0$.
\end{vetaN}

\begin{vetaN}{Rolleova vìta}
Nech $f$ je spojitá na $\left<a, b\right>$ a nech existuje $f'(x)\ \forall x \in (a,b)$. Nech $f(a)=f(b)$. Potom existuje $\xi \in (a,b): f'(\xi)=0$.
\end{vetaN}

\begin{vetaN}{Lagrangeova o støední hodnotì}
Nech $f$ je spojitá na $\left<a, b\right>$ a nech existuje $f'(x)\ \forall x \in (a,b)$. Potom existuje $$\xi \in (a,b): f'(\xi)=\frac{f(b)-f(a)}{b-a}$$
\end{vetaN}

\begin{vetaN}{Cauchyova o støední hodnotì}
Nech $f$ a $g$ jsou spojité na $\left<a, b\right>$, nech existuje $f'(x)\ \forall x \in (a,b)$ a nech existuje $g'(x)$ vlastní a nenulové. Potom existuje $$\xi \in (a,b): \frac{f(b)-f(a)}{g(b)-g(a)}=\frac{f'(\xi)}{g'(\xi)}$$

\begin{dukaz}
$g(a) \neq g(b)$, nebo jinak by podle Rolleovy vìty existovalo $\xi \in (a,b): g'(\xi)=0$. Definujeme $H(x)=(f(x)-f(a))(g(b)-g(a))-(g(x)-g(a))(f(b)-f(a))$. Potom $H(a)=H(b)=0$, $H$ je spojitá na $\left<a, b\right> \Rightarrow \exists H' na (a,b)$. Tedy podle Rolleovy vìty $\exists \xi: 0 = H'(\xi)=f'(\xi)(g(b)-g(a))-g'(\xi)(f(b)-f(a))$. Tedy $\frac{f(b)-f(a)}{g(b)-g(a)}=\frac{f'(\xi)}{g'(\xi)}$, nebo $g'(\xi) \neq 0$ a $g(b)-g(a)\neq 0$ 
\begin{flushright}$\square$\end{flushright}
\end{dukaz}
\end{vetaN}

\begin{vetaN}{L'Hospitalovo pravidlo}
%\csprimeson
Nech $a \in \mathbb{R}^{*}$ a funkce $f$, $g$ jsou definovány na nìjakém $P(a, \delta)$, $f, g$ mají v $P(a,\delta)$ vlastní derivaci, $\forall x \in P(a,\delta): g'(x)\not=0$ a existuje $\lim_{x \rightarrow a} \frac{f'(x)}{g'(x)}$. Nech platí i jedna z následujících podmínek:
\begin{penumerate}
        \item $\lim_{x \rightarrow a} f(x)=\lim_{x \rightarrow a} g(x)=0$
        \item $\lim_{x \rightarrow a+} |g(x)|=+\infty$
\end{penumerate}
Potom existuje i $\lim_{x \rightarrow a} \frac{f(x)}{g(x)}$ a platí $\lim_{x \rightarrow a} \frac{f(x)}{g(x)}=\lim_{x \rightarrow a} \frac{f'(x)}{g'(x)}$.
\end{vetaN}

\begin{vetaN}{O limitì derivací}
Nech funkce $f$ je spojitá zprava v $a \in \mathbb{R}$ a nech existuje $\lim_{x \rightarrow a_+} f'(x)=A \in \mathbb{R}^*$. Potom $f'_+(a)=A$.
\end{vetaN}

\begin{vetaN}{Vztah derivace a monotonie}
Nech $I$ je nezdegenerovanı interval (tj. nejde o jedinı bod) a $Int(I)=\{\textit{vnitøní body I}\}$. Nech $f$ je spojitá na $I$ a existuje $f'$ vlastní na $Int(I)$.:
\begin{penumerate}
        \item je-li $f' > 0$ na $Int(I)$, pak f je rostoucí na I
        \item je-li $f' \ge 0$ na $Int(I)$, pak f je neklesající na I
        \item je-li $f' < 0$ na $Int(I)$, pak f je klesající na I
        \item je-li $f' \le 0$ na $Int(I)$, pak f je nerostoucí na I
\end{penumerate}
\end{vetaN}

\begin{definiceN}{Teèna, inflexe}
Nech funkce $f$ má v $a \in \mathbb{R}$ vlastní derivaci. Oznaèíme $T_a=\{[x,y]\in \mathbb{R}^2, y=f(a)+f'(a)(x-a)\}$. Øekneme, e $[x, f(x)] \in \mathbb{R}^2$ leí nad (pod) teènou $T_a$, jestlie $f(x)>f(a)+f'(a)(x-a)$ ($f(x)<f(a)+f'(a)(x-a)$).

Nech $f$ má v $a \in \mathbb{R}$ vlastní derivaci. Øekneme, e $f$ má v $a$ \emph{inflexi}, jestlie $\exists \delta > 0$: buï $\forall x \in (a-\delta, a): [x, f(x)]$ leí nad $T_a$ $\wedge$ $\forall x \in (a, a+\delta): [x, f(x)]$ leí pod $T_a$, nebo opaènì.
\end{definiceN}

\begin{vetaN}{Nutná podmínka existence inflexe}
Jestlie $f''(a) \neq 0$, pak $f$ nemá v $a$ inflexi.
\end{vetaN}

\begin{vetaN}{Postaèující podmínka existence inflexe}
Nech $f$ má spojitou první derivaci na $(a,b)$. Nech $z \in (a,b)$. Nech $\forall x \in (a,z): f''(x)>0$ a $\forall x \in (z, b): f''(x)<0$ (nebo naopak). Pak $z$ je bod inflexe $f$.
\end{vetaN}

\begin{definice} % definice se x_1,x_2,x_3 je podle me srozumitelnejsi, ale jak myslis ... -- Tuetschek
Øekneme, e funkce $f$ je na intervalu $I$:
\begin{pitemize}
        \item \emph{konvexní}, jestlie pro kadé $x_1, x_2 \in I$ a kadé $\lambda \in \left<0, 1\right>$ platí\\$f(\lambda x_1 + (1-\lambda)x_2) \le \lambda \cdot f(x_1)+(1-\lambda)\cdot f(x_2)$.
        \item \emph{konkávní}, jestlie pro kadé $x_1, x_2 \in I$ a kadé $\lambda \in \left<0, 1\right>$ platí\\$f(\lambda x_1 + (1-\lambda)x_2) \ge \lambda \cdot f(x_1)+(1-\lambda)\cdot f(x_2)$.
        \item \emph{ryze konvexní}, jestlie pro kadé $x_1, x_2 \in I, x_1 \neq x_2$ a kadé $\lambda \in (0, 1)$ platí\\$f(\lambda x_1 + (1-\lambda)x_2) < \lambda \cdot f(x_1)+(1-\lambda)\cot f(x_2)$.
        \item \emph{ryze konkávní}, jestlie pro kadé $x_1, x_2 \in I, x_1 \neq x_2$ a kadé $\lambda \in (0, 1)$ platí\\$f(\lambda x_1 + (1-\lambda)x_2) > \lambda \cdot f(x_1)+(1-\lambda)\cdot f(x_2)$.
\end{pitemize}
\end{definice}

\begin{veta}
Nech funkce $f$ je konvexní na $I$ a $a \in Int(I)$. Potom $\exists f'_+(a) \in \mathbb{R}$ a $\exists f'_-(a) \in \mathbb{R}$. Tj. konvexnost implikuje existenci vlastních jednostrannıch derivací, neznamená to ale, e existuje derivace.
\end{veta}

\begin{vetaN}{Vztah konvexity a spojitosti}
Nech $f$ je konvexní na otevøeném intervalu $(a, b)$. Pak $f$ je na $(a, b)$ spojitá.
\end{vetaN}

\begin{veta}
Nech $f$ má spojitou první derivaci na $I=(a, b)$. Potom:
\begin{pitemize}
        \item $f''(x)>0\ \forall x \in (a, b)$, pak f je ryze konvexní na $(a, b)$
        \item $f''(x) \ge 0\ \forall x \in (a, b)$, pak f je konvexní na $(a, b)$
        \item $f''(x)<0\ \forall x \in (a, b)$, pak f je ryze konkávní na $(a, b)$
        \item $f''(x) \le 0\ \forall x \in (a, b)$, pak f je konkávní na $(a, b)$
\end{pitemize}
\end{veta}

\begin{definiceN}{Asymptota}

\end{definice}

\end{veta}

\begin{vetaN}{Obecnı tvar zbytku}
Nech $f$ má vlastní (n+1)-ní derivaci v intervalu $\left<a,x\right>, x>a$. Nech $\varphi$ je spojitá funkce na $\left<a,x\right>$, která má na $(a,x)$ vlastní nenulové derivace. Pak:
$$\exists \xi \in (a, x): R^{f,a}_{n}(x)=f(x)-T^{f,a}_n(x)=\frac{1}{n!}.\frac{\varphi(x)-\varphi(a)}
{\varphi'(\xi)}.f^{(n+1)}(\xi).(x-\xi)^n$$

$$\frac{-\frac{f^{(n+1)}(\xi)}{n!}(x-\xi)^n}{\varphi'(\xi)} = \frac{F'(\xi)}{\varphi'(\xi)} = \frac{F(x)-F(a)}{\varphi(x)-\varphi(a)} = \frac{0-R_n^{f,a}(x)}{\varphi(x)-\varphi(a)}$$

\begin{flushright}$\square$\end{flushright}

\begin{dusledek}
\emph{Lagrangeùv tvar zbytku}:

$$R^{f,a}_{n}(x)=\frac{1}{(n+1)!} f^{(n+1)}(\xi) (x-a)^{n+1}$$

$$R^{f,a}_{n}(x)=\frac{1}{n!} f^{(n+1)}(\xi) (x-\xi)^{n} (x-a)$$
\end {dusledek}
