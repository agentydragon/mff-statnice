\subsection{Definície}
\begin{e}{Definice}{0}{0}
Nech $(T,+,\cdot)$ je teleso a $V$ je mnoina (jej prvky nazıvame \emph{vektory}) s binárnou operáciou $+$ a $\cdot :T \times V\to V$ je zobrazenie, potom $(V,+,\cdot)$ sa nazıva \textbf{vektorovı prostor} nad telesom $T$ ak je splnenıch nasledujúcich 8 axiomov.
\begin{penumerate}
\item[(SA)] $\forall u,v,w \in V: \quad (u+v)+w = u+(v+w)$ \hfill\textit{(asociativita súètu)}
\item[(SK)] $\forall u,v \in V: \quad u+v = v+u$ \hfill\textit{(komutativita súètu)}
\item[(S0)] $\exists {\bf 0}\in V: \quad u + {\bf 0} = {\bf 0} + u = u$ \hfill\textit{(neutrálnı prvok súètu)}
\item[(SI)] $\forall u \in V \, \exists -u \in V: \quad u + (-u) = {\bf 0}$ \hfill\textit{(inverznı prvok súètu)}
\item[(NA)] $\forall a,b \in T \, \forall u \in V: \quad (a\cdot b) \cdot u = a \cdot (b \cdot u)$ \hfill\textit{(asociativita súèinu)}
\item[(N1)] $\forall u \in V: \quad 1 \cdot u = u$ \hspace{0.5cm} kde $1 \in T$ je jednotkovı prvok telesa T \hfill\textit{(neutrální prvek souèinu)}
\item[(D1)] $\forall a,b \in T \, \forall u \in V: \quad (a+b) \cdot u = a \cdot u + b \cdot u$ \hfill\textit{(distributivita)}
\item[(D2)] $\forall a \in T \, \forall u,v \in V: \quad a \cdot (u+v) = a \cdot u + a \cdot v$ \hfill\textit{(distributivita)}
\end{penumerate}
\end{e}

\begin{pitemize}
\item $\{{\bf 0}\}$ \dots  triviálny vektorovı priestor
\item $T^{n}$ aritmetickı vektorovı priestor dimenzie $n$ nad telesom $T$. Ide o usporiadané $n$-tice, kde + je definované predpisom

\end{pitemize}

\begin{e}{Definice}{0}{Lineárny obal, mnoina generátorov}
Nech $V$ je vektorovı priestor nad telesom T a $X$ je podmnoina $V$, potom 
$$\mathcal{L}(X) = \bigcap\{U|X \subseteq U, U \,\mathrm{je\ podpriestor}\, V\}$$
je podpriestor $V$ generovanı $X$ nazıvanı \textbf{lineárny obal $X$}. Mnoina $X$ sa potom nazıva \emph{systém generátorov} podpriestoru $\mathcal{L}(X)$. 

Keï $\mathcal{L}(X)=V$, potom $X$ je systém generátorov vektorového priestoru $V$.
\end{e}

\begin{e}{Definice}{0}{0}
\textbf{Spojení dvou podprostorù} je podprostor
$$W_1 \oplus W_2 = \mathcal{L}(W_1 \cup W_2)$$
\end{e}

\begin{e}{Vìta}{0}{0}

$$\mathcal{L}(X) = \{w\big|w = \sum_{i=1}^{n}{a_{i}u_{i}}, n \geq 0, n \,\mathrm{\textit{koneèné}}, \forall i: \, a_{i} \in T, u_{i} \in X\}$$
Špeciálne v prípade, e $X=\emptyset$ a teda $n=0$, platí $\mathcal{L}(X)=\{0\}$.
\end{e}

\begin{e}{Definice}{0}{0}
Nech $V$ je vektorovı priestor nad telesom T, potom $n$-tica vektorov $v_{1}, \dots ,v_{n} \in V$ je \textbf{lineárne nezávislá}, ak rovnica
$$a_{1}v_{1} + a_{2}v_{2} + \dots + a_{n}v_{n} = {\bf 0}$$
má iba triviálne riešenie $a_{i} = 0$ pre všetky $i \in \{1, 2, \dots, n\}$

Nekoneèná mnoina vektorov je lineárne nezávislá, ak kadá jej koneèná podmnoina je lineárne nezávislá.
\end{e}

\begin{e}{Pozorování}{0}{0}
$X$ je lineárne nezávislá práve keï $\forall u \in X: \, u \not\in \mathcal{L}(X \setminus \{u\})$
\end{e}

\begin{e}{Definice}{0}{0}

\subsection{Veta o vımene}
\begin{e}{Lemma}{0}{o vımene}
Nech $v_{1},v_{2}, \dots ,v_{n}$ je systém generátorov priestoru $V$ a pre $u \in V$ platí $u = a_{1}v_{1} + a_{2}v_{2} + \dots +a_{n}v_{n}$, potom platí 
$$\forall i: \, a_{i} \not= 0 \Rightarrow \mathcal{L}(v_{1},v_{2}, \dots, v_{i-1}, u, v_{i+1}, \dots, v_{n}) = V$$
\textit{(Inak povedané, vektor bázy ktorı sa ``podielal'' na vytvorení vektoru $u$, mome s $u$ zameni.)}
\end{e}

\begin{e}{Dùkaz}{0}{0}
Pre ¾ubovolné $w \in V$, môme písa $w = b_{1}v_{1} + b_{2}v_{2} + \dots b_{n}v_{n}$.
Do tohoto vyjadrenia miesto $v_{i}$ dosadíme vyjadrenie $v_i$ z rovnice $u = a_{1}v_{1} + a_{2}v_{2} + \dots + a_{n}v_{n}$, èím dostaneme vyjadrenie ¾ubovolného $w$ pomocou $v_{1},v_{2}, \dots, v_{i-1}, u, v_{i+1}, \dots, v_{n}$, z èoho vıplıva, e tieto vektory sú tie systém generátorov.
\end{e}

\begin{e}{Vìta}{0}{Steinitzova o vımene}
Nech $V$ je vektorovı priestor, $X \subseteq V$ je lineárne nezávislá a $Y \subseteq V$je koneènı systém generátorov. Potom existuje $Z \subseteq V$ také, e
\begin{pitemize}
        \item $|Z| = |Y|$
        \item $\mathcal{L}(Z) = V$
        \item $Z \setminus X \subseteq Y$
        \item $X \subseteq Z$
\end{pitemize}
\textit{(V krátkosti povedané - kadı nezávislı systém vektorov $X$ je moné, pridaním vektorov zo systému generátorov $Y$, rozšíri na systém generátorov $V$.)}
\end{e}

\begin{e}{Dùkaz}{0}{0}

Inak vezmeme $Y$ a postupne do neho zaèneme pridáva prvky z $X \setminus Y$.

(Pri kadej iterácií vyhadzujeme jeden prvok, ktorı nepatrí do $X$, pretoe $X$ je lineárne nezávislá.)
\end{e}

\begin{e}{Dùsledek}{0}{0}
Ak má $V$ koneènú bázu, majú všetky bázy rovnakú ve¾kos.
\end{e}

\begin{e}{Dùsledek}{0}{0}

\subsection{Lineárne zobrazenie}

\begin{e}{Definice}{0}{lineární zobrazení}
Mìjme vektorové prostory $V,W$. Øekneme, e zobrazení $f: V \to W$ je \textbf{lineární}, jestlie pro libovolná $x,y \in V$ a $a, b \in T$ platí
$$f(a \cdot x + b \cdot y) = a \cdot f(x) + b \cdot f(y)$$
\end{e}

\begin{e}{Definice}{0}{lineární operátor}
Lineární zobrazení $f: V \rightarrow V$ se nazıvá \emph{lineární operátor}.
\end{e}

\begin{e}{Pøíklady}{0}{0}
\begin{penumerate}
    \item Identické zobrazení $V$ na $V$ (to je pøíklad lineárního operátoru).
    \item Buï $\alpha$ pevné reálné èíslo. Zobrazení $V \to V$ dané pøedpisem $x \to \alpha x.$
    \item Derivace je lineární zobrazení z mnoiny reálnıch spojitıch funkcí $C_1(J)$ do mnoiny reálnıch funkcí $F(J).$
    \item V $\mathbb{R}^2$ jsou lineární zobrazení napø. zrcadlení $(x,y)\mapsto(-x,y)$ nebo zkosení $(x,y)\mapsto(x+y,y)$.
\end{penumerate}
\end{e}

\begin{e}{Definice}{0}{Hodnost lineárního zobrazení}
Pro lineární zobrazení $f:U\to V$ mezi dvìma vekt. prostory definujeme \emph{jádro zobrazení} ($\mathrm{Ker\ }f$) jako mnoinu $\mathrm{Ker\ }f=f^{-1}[\{0\}]$. \emph{Obraz} zobrazení $f$ ($\mathrm{Im\ }f$) je mnoina $\mathrm{Im\ }f=f[U]$. Jako \emph{hodnost zobrazení} $f$ oznaèíme èíslo $\mathrm{dim}(\mathrm{Ker\ }f)$.
\end{e}


\begin{e}{Vìta}{0}{Základní vlastnosti lineárního zobrazení}
Nech $f: V \rightarrow W$ je lineární zobrazení. Potom platí:
\begin{penumerate}
        \item $f(\mathbf{0}_V) = \mathbf{0}_W$
        \item $\mathrm{Im\ }f$ je podprostor prostoru W
        \item $\mathrm{Ker\ }(f)$ je podprostor prostoru V
        \item $f$ je prosté, právì kdy $\mathrm{Ker\ }(f) = \{0\}$
        \item je-li $\dim V = \dim W$ a je-li zobrazení $f$ prosté, potom je $f$ bijekce a inversní zobrazení $f^{-1}: W \rightarrow V$ je opìt lineární.
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{O dimenzi obrazu a jádra}
Pro $f:U\to V$ mezi dvìma vektorovımi prostory koneèné dimenze platí:
$$\mathrm{dim}(\mathrm{Ker\ }f)+\mathrm{dim}(\mathrm{Im\ }f)=\mathrm{dim}(U)$$
\end{e}

\begin{e}{Vìta}{0}{Báze urèuje lineární zobrazení}
Mìjme dány vektorové prostory $V,W$ a bázi $B=b_1,\dots,b_n$ prostoru $V$. Potom pro kadé lineární zobrazení $f:B\to W$ existuje právì jedno lineární zobrazení $g:V\to W$ takové, e $f(b_i)=g(b_i)\ \forall i\in\{1,\dots,n\}$.

\emph{Jiná formulace:} Pro libovolné vektory $y_1,\dots,y_n\in W$ existuje právì jedno lineární zobrazení $g:V\to W$ takové, e $g(b_i)=y_i\ \forall i\in\{1,\dots,n\}$.
\end{e}


\begin{e}{Definice}{0}{Isomorfismus}
Lineární zobrazení se nazıvá \emph{isomorfismus}, existuje-li k nìmu inversní lineární zobrazení. Pokud existuje isomorfismus $V \to W$, øíkáme, e prostory $V$ a $W$ jsou \emph{isomorfní}.
\end{e}

\begin{e}{Vìta}{0}{0}

\end{e}

\end{e}

\begin{e}{Vìta}{0}{Skládání lineárních zobrazeni}

$$ (g \circ f)(x) = g(f(x)) \,\, \textit{pro } x \in U$$

\end{e}


\begin{e}{Vìta}{0}{Sèítání a násobky lin. zobrazení}

$$(f+g)(x) = f(x) + g(x), \, x \in V$$
$$(\alpha f)(x) = \alpha f(x), \, x \in V$$

\end{e}

\begin{e}{Vìta}{0}{Mnoina lineárních zobrazení je vekt. prostor}
