\subsection{Primitivní funkce, metody vıpoètu}

\begin{e}{Definice}{0}{0}
Nech funkce $f$ je definována na otevøeném intervalu $I$. Øekneme, e funkce $F$ je \emph{primitivní funkce k $f$ na $I$}, jestlie pro kadé $x \in I$ existuje $F'(x)$ a platí $F'(x)=f(x)$.
\end{e}

\begin{e}{Vìta}{0}{Tvar primitivní funkce}
Nech $F$ a $G$ jsou dvì primitivní funkce k funkci $f$ na otevøeném intervalu $I$. Pak existuje $c \in \mathbb{R}$ tak, e $F(x)=G(x)+c$ pro kadé $x \in I$.
\end{e}

\begin{e}{Vìta}{0}{Linearita primitivní funkce}
Nech f má na otevøeném intervalu $I$ primitivní funkci $F$, funkce $g$ má na na $I$ primitivní funkci $G$ a $\alpha, \beta \in \mathbb{R}$. Potom funkce $\alpha F + \beta G$ je primitivní funkcí k $\alpha f + \beta g$ na $I$.

\begin{e}{Poznámka}{0}{0}
Pøedchozí tvrzení èasto zapisujeme (pokud alespoò jedno z èísel $\alpha, \beta$ je rùzné od nuly)
$$\int (\alpha f(x) + \beta g(x)) \d x = \alpha \int f(x) \d x + \beta \int g(x) \d x.$$
\end{e}
\end{e}

\begin{e}{Vìta}{0}{Spojitost a existence primitivní funkce}
Nech $f$ je spojitá funkce na otevøeném intervalu $I$. Pak $f$ má na $I$ primitivní funkci.
\end{e}

\begin{e}{Vìta}{0}{O substituci}
\begin{penumerate}
\item Nech $F$ je primitivní funkce k $f$ na $(a,b)$. Nech $\varphi$ je funkce definována na $(\alpha, \beta)$ s hodnotami v intervalu $(a,b)$, která má v kadém bodì $t \in (\alpha, \beta)$ vlastní derivaci. Pak
$$\int f(\varphi(t))\varphi'(t)dt=F(\varphi(t))+C \textit{ na } (\alpha,\beta)$$
\item Nech funkce $\varphi$ má v kadém bodì intervalu $(\alpha,\beta)$ nenulovou vlastní derivaci a $\varphi((\alpha,\beta)) = (a,b)$. Nech funkce $f$ je definována na intervalu $(a,b)$ a platí
$$\int f(\varphi(t))\varphi'(t)dt=G(t)+C \textit{ na } (\alpha,\beta)$$
Pak
$$\int f(x)\d x=G(\varphi^{-1}(x))+C \textit{ na } (a,b)$$
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{Integrace per partes}
Nech $I$ je otevøenı interval a funkce $f$ a $g$ jsou spojité na $I$. Nech $F$ je primitivní funkce k $f$ na $I$ a $G$ je primitivní funkce ke $g$ na $I$. Pak platí
$$\int g(x)F(x)\d x=G(x)F(x) - \int G(x)f(x)\d x \textit{ na } I$$

(\emph{Poznámka autora}: $\int u'v = uv - \int uv'$)
\end{e}

\subsubsection{Postup integrace racionální funkce}

\begin{e}{Vìta}{0}{Dìlení polynomu}
Nech P a Q jsou polynomy s reálnımi koeficienty, pøièem Q není identicky roven nule. Pak existují (jednoznaènì urèené) polynomy R a S takové, e stupeò S je menší ne stupeò Q a pro všechna $x \in \mathbb{R}$ platí $P(x)=R(x)Q(x)+S(x)$.
\end{e}

\begin{e}{Vìta}{0}{Základní vìta algebry}
Nech $P(x)=a_nx^n+\dots+a_1x+a_0$ je polynom stupnì $n$ s reálnımi koeficienty. Pak existují èísla $x_1,\dots,x_n \in \mathbb{C}$ taková, e
$$P(x)=a_n(x-x_1)\dots(x-x_n), \,\, x \in \mathbb{R}$$
\end{e}

\begin{e}{Vìta}{0}{O koøenech polynomu}
Nech $P$ je polynom s reálnımi koeficienty a $z \in \mathbb{C}$ je koøen $P$ násobnosti $k \in \mathbb{N}$. Pak i $\overline{z}$ je koøen $P$ násobnosti k.
\end{e}

\begin{e}{Vìta}{0}{O rozkladu polynomu}
Nech $P(x)=a_nx^n +\dots+a_1x+a_0$ je polynom stupnì $n$ s reálnımi koeficienty. Pak existují reálná èísla $x_1,\dots,x_k,\alpha_1,\dots,\alpha_l,\beta_1,\dots,\beta_l$ a pøirozená èísla $p_1,\dots,p_k,q_1,\dots,q_l$ taková, e:

\begin{penumerate}
	\item $P(x)=a_n(x-x_1)^{p_1} \dots (x-x_k)^{p_k}(x^2+\alpha_1x+\beta_1)^{q_1}\dots(x^2+\alpha_lx+\beta_l)^{q_l}$,
	\item ádné dva z mnohoèlenù $x-x_1,x-x_2,\dots,x-x_k,x^2+\alpha_1x+\beta_1,\dots,x^2+\alpha_lx+\beta_l$ nemají spoleènı koøen,
	\item mnohoèleny $x^2+\alpha_1x+\beta_1,\dots,x^2+\alpha_lx+\beta_l$ nemají ádnı reálnı koøen.
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{O rozkladu na parciální zlomky}
Nech $P,Q$ jsou polynomy s reálnımi koeficienty takové, e

\begin{penumerate}
	\item stupeò $P$ je ostøe menší ne stupeò $Q$,
	\item $Q(x)=a_n(x-x_1)^{p_1}\dots(x-x_k)^{p_k} (x^2+\alpha_1x+\beta_1)^{q_1}\dots(x^2+\alpha_lx+\beta_l)^{q_l}$,
	\item $a_n,x_1,\dots,x_k,\alpha_1,\dots,\alpha_l,\beta_1,\dots,\beta_l \in \mathbb{R}, a_n \neq 0$
	\item $p_1,\dots,p_k,q_1,\dots,q_l \in \mathbb{N}$
	\item ádné dva z mnohoèlenù $x-x_1, x-x_2,\dots,x-x_k,x^2+\alpha_1x+\beta_1,\dots,x^2+\alpha_lx+\beta_l$ nemají spoleènı koøen
	\item mnohoèleny $x^2+\alpha_1x+\beta_1, \dots, x^2+\alpha_lx+\beta_l$ nemají reálnı koøen
\end{penumerate}

Pak existují jednoznaènì urèená èísla $A_1^1,\dots,A_{p_1}^{1},\dots,A_1^k,\dots,A_{p_k}^k,\\B_1^1,C_1^1,\dots,B_{q_1}^1,C_{q_1}^{1},\dots,B_1^l,C_1^l,\dots,B_{q_l}^l,C_{q_l}^l$ taková, e platí


\begin{align*}
\frac{P(x)}{Q(x)} & = \frac{A_1^1}{(x-x_1)^{p_1}} + \dots + \frac{A_{p_1}^1}{(x-x_1)} + \dots + \frac{A_1^k}{(x-x_k)^{p_k}} + \dots + \frac{A_{p_k}^k}{(x-x_k)}\\
&+ \frac{B_1^1x+C_1^1}{(x^2+\alpha_1x+\beta_1)^{q_1}} + \dots + \frac{B_{q_1}^1x+C_{q_1}^1}{x^2+\alpha_1x+\beta_1} + \dots\\
&+ \frac{B_1^lx+C_1^l}{(x^2+\alpha_lx+\beta_l)^{q_l}} + \dots + \frac{B_{q_l}^lx+C_{q_l}^l}{x^2+\alpha_lx+\beta_l}.
\end{align*}

\end{e}

\textbf{Postup integrace racionální funkce $\frac{P(x)}{Q(x)}$ je:}
\begin{penumerate}
\item Vydìlíme polynomy $P$ a $Q$ - najdeme polynomy $R$ a $S$ takové, e
	$$\frac{P(x)}{Q(x)} = R(x) + \frac{S(x)}{Q(x)}$$
	a stupeò S je menší ne stupeò Q.

\item Najdeme rozklad polynomu Q ve tvaru uvedeném ve \emph{vìtì o rozkladu polynomu}.

\item Najdeme rozklad $\frac{S}{Q}$ na parciální zlomky ve tvaru uvedeném ve \emph{vìtì o rozkladu na parciální zlomky}.

\item Najdeme primitivní funkce ke všem parciálním zlomkùm.
\end{penumerate}

\subsection{Urèitı (Riemannùv) integrál, uití urèitého integrálu}

\begin{e}{Definice}{0}{0}
Koneènou posloupnost $D=\{x_j\}_{j=0}^n$ nazıváme \emph{dìlením intervalu} $\left<a,b\right>$, jestlie platí
$$a=x_0 < x_1 < \dots < x_n = b$$

Body $x_0,\dots,x_n$ nazıváme dìlícími body. \emph{Normou dìlení} $D$ rozumíme èíslo
$$\upsilon(D)=\max\{x_j-x_{j-1}; j=1,\dots,n\}.$$

Øekneme, e dìlení $D'$ intervalu $\left<a,b\right>$ je \emph{zjemnìním dìlení} $D$ intervalu $\left<a,b\right>$, jestlie kadı dìlící bod $D$ je i dìlícím bodem $D'$.
\end{e}

\begin{e}{Definice}{0}{0}
Nech $f$ je omezená funkce definovaná na intervalu $\left<a,b\right>$ a $D=\{x_j\}_{j=0}^n$ je dìlení $\left<a,b\right>$. Oznaème
$$S(f,D)=\sum_{j=1}^{n} M_j(x_j-x_{j-1}), \textit{ kde } M_j=\sup\{f(x); x\in \left<x_{j-1},x_j\right>\}$$
$$s(f,D)=\sum_{j=1}^{n} m_j(x_j-x_{j-1}), \textit{ kde } m_j=\inf\{f(x); x\in \left<x_{j-1},x_j\right>\}$$

\begin{e}{Poznámka}{0}{0}
Nech $f$ je omezená funkce definovaná na intervalu $\left<a,b\right>$.
\begin{penumerate}
	\item Pro kadé dìlaní $D$ intervalu $\left<a,b\right>$. platí $s(f,D)\le S(f,D)$.
	\item Je-li $D_1$ zjemnìním $D_2$, pak $s(f,D_1) \ge s(f,D_2)$ a $S(f,D_1) \le S(f, D_2)$
	\item Jsou-li $D_1$ a $D_2$ dìlení intervalu $\left<a,b\right>$, pak $s(f,D_1) \le S(f, D_2)$.
\end{penumerate}
\end{e}
\end{e}

\begin{e}{Definice}{0}{0}
Nech $f$ je omezená funkce definovaná na intervalu $\left<a,b\right>$.
\begin{penumerate}
\item
Oznaème
$$\overline{\int_a^b}f=\inf\{S(f,D); D \textit{ je dìlením intervalu } \left<a,b\right>\}$$
\begin{center}(tzv. horní \emph{Riemannùv integrál funkce} $f$ pøes $\left<a,b\right>$),\end{center}

$$\underline{\int_a^b}f=\sup\{s(f,D); D \textit{ je dìlením intervalu } \left<a,b\right>\}$$
\begin{center}(tzv. dolní \emph{Riemannùv integrál funkce} $f$ pøes $\left<a,b\right>$)\end{center}

\item Øekneme, e funkce $f$ má \emph{Riemannùv integrál pøes} $\left<a,b\right>$, pokud $\overline{\int_a^b}f=\underline{\int_a^b}f$. Hodnota tohoto integrálu je pak rovna $\overline{\int_a^bf}$ a znaèíme ji $\int_a^bf$. 

Pokud $a>b$, definujeme $\int_a^bf=-\int_b^af$, v pøípadì, e $a=b$, definujeme $\int_a^b=0$.
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{Kritérium existence Riemannova integrálu}
Nech $a<b$ a $f$ je funkce omezená na $\left<a,b\right>$. Pak $\int_a^bf$ existuje, právì kdy pro kadé $\varepsilon > 0$ existuje dìlení $D$ intervalu $\left<a,b\right>$ takové, e $S(f,D)-s(f,D)<\varepsilon$
\end{e}

\begin{e}{Vìta}{0}{Monotonie a linearita Riemannova integrálu}
Nech $a,b\in \mathbb{R}, a<b$ a funkce $f,g$ mají Riemannùv integrál pøes interval $\left<a,b\right>$. Pak platí:
\begin{penumerate}
	\item Jestlie pro kadé $x \in \left<a,b\right>$ je $f(x) \le g(x)$, pak $\int_a^bf \le \int_a^bg$
	\item $\int_a^b(f+g) = \int_a^bf + \int_a^bg$
	\item $\int_a^bcf=c\int_a^bf$ pro kadé $c \in \mathbb{R}$
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{Spojitost a Riemannovská integrovatelnost}
Nech $a,b\in \mathbb{R}, a<b$ a funkce $f$ je spojitá na intervalu $\left<a,b\right>$. Pak existuje $\int_a^bf$.
\begin{e}{Poznámka}{0}{0}
Platí dokonce: Pokud je $f$ omezená na $\left<a,b\right>$ a je spojitá ve všech bodech intervalu $\left<a,b\right>$ s vıjimkou koneènì mnoha, pak existuje $\int_a^bf$.
\end{e}
\end{e}


\begin{e}{Vìta}{0}{Monotonie a Riemannovská integrovatelnost}
Je li $f$ omezená a monotónní na uzavøeném intervalu, pak je Riemannovsky integrovatelná. % Proc ty otazniky? to plati. -- Tuetschek
\end{e}

\begin{e}{Vìta}{0}{Vlastnosti $\int$}
Nech $a,b\in \mathbb{R}, a<b$ a funkce $f$ je omezená na intervalu $\left<a,b\right>$. Pak platí
\begin{penumerate}
	\item Jestlie existuje $\int_a^bf$, pak pro kadı interval $\left<c,d\right> \subset \left<a,b\right>$ existuje $\int_c^df$.
	\item Je-li $c \in (a,b)$, pak $\int_a^bf=\int_a^cf+\int_c^bf$, má-li alespoò jedna strana smysl (\emph{aditivita Riemannova integrálu jako funkce intervalu})
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{Riemannùv integrál jako primitivní funkce}
Nech $a,b\in \mathbb{R}, a<b$ a funkce $f$ je omezená na intervalu $\left<a,b\right>$. Pro $x \in \left<a,b\right>$ polome $F(x)=\overline{\int_a^x}f$. Potom platí:
\begin{penumerate}
	\item Funkce $F$ je spojitá na $\left<a,b\right>$
	\item Je-li $x_0 \in (a,b)$ a funkce $f$ je v bodì $x_0$ spojitá, pak $F'(x_0)=f(x_0)$.
\end{penumerate}
Stejná tvrzení platí i pro funkci $G(x)=\underline{\int_a^x}f$
\end{e}

\begin{e}{Poznámka}{0}{0}
Pro Riemannovy integrály lze pouít i metodu per partes nebo pravidlo substituce. % tohle by mozna chtelo formulovat nejak formalneji -- Tuetschek
\end{e}


\begin{e}{Vìta}{0}{Základní vìta analızy}
Nech $a,b\in \mathbb{R}, a<b$ a funkce $f$ je spojitá na intervalu $\left<a,b\right>$. Nech $F$ je primitivní funkce k $f$ na intervalu $(a,b)$. Pak existují vlastní limity $\lim_{x\rightarrow a+}F(x)$ a $\lim_{x\rightarrow b-}F(x)$ a platí:
$$\int_a^bf = (\lim_{x\rightarrow b-}F(x))-(\lim_{x\rightarrow a+}F(x))$$
\end{e}


\begin{e}{Definice}{0}{Newtonùv integrál}
Nech funkce $f$ je definována na intervalu $(a,b)$ a $F$ je primitivní funkce k $f$ na $(a,b)$. \emph{Newtonovım integrálem} funkce $f$ pøes interval $(a,b)$ nazıváme èíslo
$$(N)\int_a^bf = (\lim_{x\rightarrow b-}F(x))-(\lim_{x\rightarrow a+}F(x))$$
pokud obì limity na pravé stranì existují a jsou vlastní.
\end{e}

\begin{e}{Poznámka}{0}{Vztah Riemannova a Newtonova integrálu}
Je-li funkce $f$ spojitá na intervalu $\left<a,b\right>$, pak platí: 
$$(N)\int_a^b f(x) \d x = (R)\int_a^b f(x) \d x$$

Mnoiny funkcí integrovatelnıch newtonovsky a riemannovsky jsou neporovnatelné.
\end{e}

\subsection{Vícerozmìrnı integrál a Fubiniho vìta}

\begin{e}{Definice}{0}{0}
(Kompaktním) \emph{intervalem} v $n$-rozmìrném euklidovském prostoru $E_n$ rozumíme souèin
$$J = \left<a_1,b_1\right> \times \dots \times \left<a_n,b_n\right>$$
kde $\left<a_i,b_i\right>$ jsou kompaktní intervaly v $\mathbb{R}$.
\end{e}

\begin{e}{Definice}{0}{0}
\begin{pitemize}
\item \emph{Rozdìlením} $D$ takového intervalu $J$ rozumíme $n$-tici $D_1, \dots, D_n$, kde $D_i$ je rozdìlení intervalu $\left<a_i,b_i\right>$.
\item Rozdìlení $D=(D_1,\dots,D_n)$ je \emph{zjemnìním} $D'=(D'_1,\dots,D'_n)$ jestlie $D_i$ zjemòuje $D'_i$.
\end{pitemize}
\end{e}

\begin{e}{Pozorování}{0}{0}
Kadé dvì rozdìlení mají spoleèné zjemnìní.
\end{e}

\begin{e}{Definice}{0}{0}
\emph{Èlen rozdìlení} $D=(D_1,\dots,D_n)$ je kterıkoliv interval $K=\left<t_{1, i_1}, t_{1, i_1+1}\right> \times \dots \times \left<t_{n, i_n}, t_{n, i_n+1}\right>$, kde $D_k: t_{k0} < \dots < t_{k,r(k)}, \, \, 0 \le i_j \le r(j)$. Mnoina všech èlenù rozdìlení $D$ bude oznaèována $|D|$. % urcite tu melo byt podrozdeleni?? -- Tuetschek
\end{e}

\begin{e}{Definice}{0}{0}
\emph{Objem intervalu} $J=\left<a_1,b_1\right> \times \dots \times \left<a_n,b_n\right>$ je èíslo
$$\vol J=(b_1-a_1).(b_2-a_2)\dots(b_n-a_n)$$
\end{e}

\begin{e}{Definice}{0}{0}
Buï $f$ omezená funkce na intervalu $J$, buï $D$ rozdìlení $J$. \emph{Dolní (resp. horní) sumou} funkce $f$ v rozdìlení $D$ rozumíme èíslo
$$s(f,D)=\sum_{K\in |D|}m_K\cdot\vol K \textit{\;\;resp.\;\;} S(f,D) = \sum_{K \in |D|} M_K\cdot\vol K,$$
kde $m_K$ je infimum a $M_k$ supremum funkce $f$ na intervalu $K$.
\end{e}

\begin{e}{Pozorování}{0}{0}
Pro libovolná dvì rozdìlení $D$ a $D'$ platí $s(f,D) \le S(f,D')$
\end{e}

\begin{e}{Definice}{0}{0}
Dolní a horní Riemannùv integrál definujeme jako
$$\underline{\int}_Jf=\sup_Ds(f,D), \;\;\; \overline{\int}_Jf=\inf_DS(f,D)$$
a pøi rovnosti tìchto hodnot mluvíme o Riemannovì integrálu a píšeme prostì
$$\int_Jf \textit{ nebo } \int_Jf(x_1,\dots,x_n)\d x_1\dots x_n, \;\;\;\int_Jf(\overrightarrow{x})d\overrightarrow{x}.$$
\end{e}

\begin{e}{Vìta}{0}{0}
Pro vícerozmìrnı Riemannùv integrál platí (podobnì jako pro jednorozmìrnı pøípad) e $f$ je Riemannovsky integrovatelná, právì kdy ke kadému $\varepsilon > 0$ existuje rozdìlení $D$ takové, e
$$S(f,D)-s(f,D)<\varepsilon$$

Platí i vìta, e spojitá funkce na intervalu $J$ je Riemannovsky integrovatelná.
\end{e}

\begin{e}{Vìta}{0}{Vlastnosti Riemannova vícerozmìrného integrálu}
Platí:
\begin{penumerate}
	\item $|\int_J f| \le \int_J|f|$ (existují-li pøíslušné integrály)
	\item Buïte $f,g$ Riemannovsky integrovatelné funkce na $J$, buï $f \le g$. Potom $\int_J f \le \int_J g$.
	\item Speciálnì, je-li $f(\overrightarrow{x})\le C$ pro nìjakou konstantu $C$, platí $\int_J f \le C\cdot \vol J$
\end{penumerate}
\end{e}

\begin{e}{Vìta}{0}{Fubiniova}
Buïte $J' \subseteq E^n, J'' \subseteq E^m$ intervaly, $J=J' \times J''$, buï $f$ spojitá funkce definovaná na $J$. Potom
$$\int_J f(\overrightarrow x,\overrightarrow y)d\overrightarrow x\overrightarrow y =
\int_{J'}\left(\int_{J''}f(\overrightarrow x,\overrightarrow y)d\overrightarrow y\right)d\overrightarrow x =
\int_{J''}\left(\int_{J'}f(\overrightarrow x,\overrightarrow y)d\overrightarrow x\right)d\overrightarrow y
$$

(Inımi slovami: Hodnota \uv{integrálu} cez celı interval je rovná hodnote po integrovaní postupne cez (jednotlivé) \uv{rozmery} - prièom je moné integrova v ¾ubovo¾nom poradí.)
\end{e}

\begin{e}{Definice}{0}{Parciální derivace}
\emph{Parciální derivace} funkce $f$ v bodì $a\in\mathbb{R}^n$ podle promìnné $x_i$ se definuje následovnì:
$$\frac{\partial f}{\partial x_i}(a)=\lim_{h\to 0}\frac{f(a_1,\dots,a_{i-1},a_i + h,a_{i+1},\dots,a_n)-f(a_1,\dots,a_n)}{h}$$
\end{e}

\begin{e}{Definice}{0}{Jacobiho matice, jakobián}
\emph{Jacobiho matice} funkce $\overrightarrow f:D\to \mathbb{R}^n$ v bodì $a\in D$, kde $D$ je otevøená mnoina v $\mathbb{R}^m$ a $f_1,f_2,\dots f_n$ jsou souøadnicové funkce $f$, je dána pøedpisem:
$$
\left(\frac{\partial f_i}{\partial x_j}(a)\right)_{i,j=1}^{n,m} =
\left( \begin{matrix}
	\frac{\partial f_1}{\partial x_1} & \cdots & \frac{\partial f_1}{\partial x_m} \\
	\vdots & \ddots & \vdots \\
	\frac{\partial f_n}{\partial x_1} & \cdots & \frac{\partial f_n}{\partial x_m}
\end{matrix} \right)
$$

\noindent Táto ètvercová matice se obvykle znaèí $\frac{D(f_1,\dots,f_n)}{D(x_1,\dots,x_n)}$ -- a je-li $m=n$, její determinant se nazıvá \emph{Jakobián} (a znaèí se rovnako???).
\end{e}

\begin{e}{Definice}{0}{Regulární zobrazení}
Nech $U \subseteq \mathbb{R}^n$ je otevøená mnoina, $\overrightarrow f: U \rightarrow \mathbb{R}^n$ má spojité parciální derivace. \emph{Zobrazení $\overrightarrow f$ je regulární}, je-li jakobián
$$\frac{D(f_1,\dots,f_n)}{D(x_1,\dots,x_n)}(\overrightarrow x) \neq 0, \,\, \forall \overrightarrow x \in U$$
\end{e}


\begin{e}{Vìta}{0}{O substituci}
Nech $\varphi: U \subseteq \mathbb{R}^n \rightarrow \mathbb{R}^n$ je regulární zobrazení, $A$ je uzavøená mnoina v $\mathbb{R}^n$, $A \subseteq U$ na které existuje $\int_{\varphi(A)}f(\overrightarrow x)d\overrightarrow x$. Potom platí:
$$\int_A f(\overrightarrow{\varphi}(\overrightarrow t)) \frac{D(\overrightarrow \varphi)}{D(\overrightarrow t)}d\overrightarrow t = \int_{\varphi (A)} f(\overrightarrow x)d\overrightarrow x$$
\end{e}
