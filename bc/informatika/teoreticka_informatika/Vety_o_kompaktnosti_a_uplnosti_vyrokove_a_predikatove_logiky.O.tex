\subsection{Vìty o kompaktnosti a úplnosti vırokové a predikátové logiky}

\begin{e}{Definice}{0}{úplná teorie}
Teorie $T$ s jazykem $L$ prvního øádu je \emph{úplná}, je-li bezesporná a pro libovolnou uzavøenou formuli $A$ je jedna z formulí $A,\neg A$ dokazatelná v $T$.
\end{e}

\begin{e}{Vìta}{1}{o korektnosti\vl}
Pro teorii $T$ a formuli $A$ platí $T \vdash A \Rightarrow T \models A$. (Kadá v $T$ dokazatelná formule je v $T$ pravdivá.)
\end{e}
\begin{e}{Dùkaz}{0}{0}
Indukcí na vìtách $T$. Kadı axiom je pravdivı (ovìøíme pøímo) a pravidlo modus ponens také zachovává pravdivost.
\end{e}

\begin{e}{Vìta}{0}{o bezespornosti a modelech\vl}
Má-li teorie model, je bezesporná.
\end{e}
\begin{e}{Dùkaz}{0}{0}
Formule $A$ a $\neg A$ nemohou bıt zároveò platné v ádném modelu.
\end{e}

\begin{e}{Vìta}{1}{o spoustì vìcí\vl}
Nech $A, B$ jsou formule teorie $T$. Platí následující tvrzení.
\begin{penumerate}
    \item
    \begin{penumerate}
        \item Teorie $T$ je sporná, právì kdy je v ní dokazatelnı spor.
        \item (Dùkaz sporem.) $T, \neg A$ je sporná právì kdy $T \vdash A$.
    \end{penumerate}
    \item Buï $T$ maximální bezesporná teorie. Pak platí:
    \begin{penumerate}
        \item $T \vdash A \Leftrightarrow A \in T \Leftrightarrow T, A$ je bezesporná
        \item $A \in T \Leftrightarrow \neg A \notin T$ a dále $\left(A \to B\right) \in T \Leftrightarrow (\neg A \in T$ nebo $B \in T)$.
        \item Ohodnocení $e$ takové, e $e(p) = 1 \Leftrightarrow p \in T$ pro kadou vırokovou promìnnou $p$, je jedinı model $T$.
    \end{penumerate}
    \item Bezesporná teorie má maximální bezesporné rozšíøení (v téme jazyce).
    \item (\textbf{O existenci modelu}.) Teorie má model, právì kdy je bezesporná.
    \item (\textbf{O kompaktnosti}.) Teorie má model (tedy je splnitelná), právì kdy kadá její koneèná podteorie má model (je splnitelná).
    \item (\textbf{O úplnosti}.) $T \vdash A \Leftrightarrow T \models A$ platí pro kadou teorii $T$ a její formuli $A$. Dùsledkem je bezespornost vırokové logiky a dokazatelné v ní jsou právì tautologie.
\end{penumerate}
\begin{e}{Dùkaz}{0}{0}
\begin{penumerate}
    \item
    \begin{penumerate}
        \item Je-li $A$ spor ($\vdash \neg A$) a pøitom $T \vdash A$, pak díky vìtì $\vdash \neg A \to (A \to B)$ platí jakıkoliv vırok $B$.
        \item
        \begin{pitemize}
            \item[$\Rightarrow$] Je-li $T, \neg A$ sporná, pak $T \vdash \neg A \to A$ uitím vìty o dedukci. Díky vìtì $\vdash (\neg A \to A) \to A$ pak platí $T \vdash A$.
            \item[$\Leftarrow$] Opìt díky vìtám o dedukci a $\vdash \neg A \to (A \to B)$ lze dokázat libovolnou formuli.
        \end{pitemize}
    \end{penumerate}
    \item 
    \begin{penumerate}
        \item z definic a maximálnosti
        \item $\neg A \notin T \Leftrightarrow T, \neg A$ je sporná $\Leftrightarrow T \vdash A \Leftrightarrow A \in T$ dle 2a) a dùkazu sporem.\\
        Tvrzení o implikaci: Kdy $A \to B \in T$, tak z $\neg A \notin T$ plyne $A \in T$. Pak $T \vdash B$ a díky 2a) je $B \in T$. Kdy $\neg A \in T$, tak $T \vdash A \to B$ díky vìtì $\vdash \neg C \to (C \to D)$, tedy $A \to B \in T$ díky a). Podobnì kdy $B \in T$, tak $T, A \vdash B$, tudí $T \vdash A \to B$.
        \item Platí $A \in T \Leftrightarrow e(A) = 1$, co plyne indukcí dle sloitosti $A$ ihned uitím b). Tedy $e \models T$. Koneènì pro $e' \models T$ máme $e'(p) = 1 \Leftrightarrow p \in  T$ pro kadou vırokovou promìnnou $p$, tedy $e' = e$.
    \end{penumerate}
    \item Plyne z principu maximality (ekvivalentního s axiomem vıbìru), aplikujeme-li jej na mnoinu všech bezespornıch teorií $S$ s $S \supseteq T$, na ní uvaujeme uspoøádání inkluzí. Kadı øetìzec $R$ v popsaném uspoøádání má majorantu, kterou je jeho sjednocení $\bigcup R$, nebo to je teorie rozšiøující $T$, která je bezesporná, protoe spor v ní je sporem v nìjaké teorii z $R$.
    \item
    \begin{pitemize}
	    \item[$\Leftarrow$]Má-li $T$ model $e$ a $T \vdash A$, tak $e(A) = 1$, tedy $e(\neg A) = 0$, tedy $T \not \vdash \neg A$ a $T$ je bezesporná.
	    \item[$\Rightarrow$]Nech je $T$ bezesporná. Dle 3) existuje maximální bezesporná teorie $T' \supseteq T$ a dle 2c) existuje model teorie $T'$, co je i model $T$.
    \end{pitemize}
    \item Plyne z 4) z toho, e teorie je bezesporná, právì kdy je bezesporná kadá její koneèná podteorie.
    \item $\Rightarrow$ je vìta o korektnosti, v opaèném smìru pøedpokládáme $T \models A$. Pak je $T, \neg A$ sporná dle tvrzení o existenci modelu 4), tedy $T \vdash A$ dle dùkazu sporem 1b).
\end{penumerate}
\end{e}
\end{e}

\begin{e}{Vìta}{0}{o ekvivalenci\vl}
Vznikne-li formule $A'$ z $A$ nahrazením nìkterého vıskytu podformule $B$ formulí $B'$, tak platí:
\begin{penumerate}
    \item $\vdash B \leftrightarrow B' \to A \leftrightarrow A'$,
    \item $T \vdash B \leftrightarrow B' \Rightarrow T \vdash A \leftrightarrow A'$.
\end{penumerate}
\end{e}

\begin{e}{Vìta}{1}{o existenci modelu, úplnosti a kompaktnosti\pl}
\begin{penumerate}
    \item (\textbf{O existenci modelu}.) Kadá bezesporná teorie má model kardinality nejvıše $\|L(T)\|$.
    \item (\textbf{O úplnosti}.) Formule teorie $T$ je dokazatelná, právì kdy je pravdivá ($T \vdash A \Leftrightarrow T \models A$).
    \item (\textbf{O kompaktnosti}.) Teorie má model (je splnitelná), právì kdy kadá její koneèná èást má model. ($T \models A$ právì kdy $T' \models A$ pro nìjakou koneènou podteorii $T' \subseteq T$.)
\end{penumerate}
\begin{e}{Dùkaz}{0}{0}
\begin{penumerate}
    \item[2.] Pro formuli $A(x)$ uitím pravidla generalizace, dùkazu sporem a vìty o existenci modelu máme: $T \not\vdash A \Leftrightarrow T \not\vdash (\forall x)A \Leftrightarrow T, (\exists x)\neg A$ je bezesporná $\Leftrightarrow T, (\exists x)\neg A$ má model $\Leftrightarrow T \not\models A$.
    \item[3.] Plyne z toho, e teorie je sporná, právì kdy je nìjaká její koneèná èást sporná.
\end{penumerate}
\end{e}
\end{e}
