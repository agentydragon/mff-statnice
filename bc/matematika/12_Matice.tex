\begin{e}{Definice}{0}{0}
Obdélníkové schéma sestavené z reálnıch èísel

\end{e}

\begin{e}{Definice}{0}{Jednotková matice}
Ètvercová matice øádu $n$ tvaru

\end{e}

\begin{e}{Definice}{0}{Nulová matice}
Ètvercovou matici $A$ typu $m\times n$, pro kterou $a_{i,j}=0\ \forall i\in\{1,\dots,m\},\forall j\in\{1,\dots,n\}$ nazveme \emph{nulová matice} a oznaèíme $\mathbf{0}$.
\end{e}



\begin{e}{Definice}{0}{Prostory související s maticí}
Buï $A$ matice typu $m\times n$ nad tìlesem $\mathbb{K}$. Potom jsou s ní spojené tyto vektorové prostory:
\begin{pitemize}
    \item \emph{sloupcovı prostor, té sloupcovı modul} -- podprostor $\mathbb{K}^m$ generovanı sloupci $A$
    \item \emph{øádkovı prostor, té øádkovı modul} -- podprostor $\mathbb{K}^n$ generovanı øádky $A$
    \item \emph{jádro matice} ($\Ker A$) -- podprostor $\mathbb{K}^n$ generovanı všemi øešeními soustavy $Ax=0$

\noindent Je zøejmé, e elementární maticové úpravy nemìní ani øádkovı prostor, ani jádro.

\begin{e}{Definice}{0}{Hodnost matice}
Hodnost matice $A$ je maximální poèet lineárnì nezávislıch sloupcù matice $A$ (jako vektorù), znaèíme ji $\rank(A)$. Hodnost matice je rovna dimenzi sloupcového prostoru (to je ekvivalentní definice).
\end{e}

\begin{e}{Vìta}{0}{O hodnosti matice}
Pro libovolnou matici $A$ typu $m\times n$ je dimenze jejího sloupcového prostoru rovna dimenzi øádkového prostoru. Tedy hodnost matice je rovna i dimenzi øádkového prostoru a platí $$\rank(A)\leq\min\{m,n\}$$

\begin{e}{Dùkaz}{0}{0}

\end{e}
\end{e}

\begin{e}{Vìta}{0}{O dimenzích maticovıch prostorù}
Pro matici $A$ s $n$ sloupci platí:
$$\dim(\Ker A)+\rank(A)=n$$
\end{e}

\begin{e}{Poznámka}{0}{0}
Po provedení \emph{Gaussovy eliminace} na matici A ($\Rightarrow A^R$) je hodnost matice A rovna poètu nenulovıch øádkù matice $A^R$.
\end{e}


\begin{e}{Definice}{0}{Regulární matice}

\subsection{Operace s maticemi a jejich vlastnosti}

\subsubsection*{Souèet a násobení skalárem}

\begin{e}{Definice}{0}{Sèítání}
Nech $A,B$ jsou matice typu $m \times n$. Potom jejich \emph{souètem} $A+B$ nazıváme matici typu $m \times n$ s koeficienty $$(A+B)_{ij} = A_{ij} + B_{ij}$$ pro $i=1,\dots,m; j=1,\dots,n$. Jsou-li $A,B$ rùznıch typù, potom souèet $A+B$ není definován.
\end{e}

\begin{e}{Definice}{0}{Násobení skalárem}
Nech $A,B$ jsou matice typu $m \times n$ a $\alpha$ skalár. Potom $\alpha\cdot A$ je matice typu $m \times n$ s koeficienty $$(\alpha\cdot A)_{ij} = \alpha\cdot A_{ij}$$ pro $i=1,\dots,m; j=1,\dots,n$. Nikdy nepíšeme $A\cdot\alpha$.
\end{e}

\begin{e}{Lemma}{0}{Vlastnosti souètu matic a násobení matic skalárem}

Nech $A,B,C$ jsou matice typu $m \times n$ a $\alpha, \beta$ skaláry. Potom platí:
\begin{penumerate}
	\item $A+B=B+A \hfill \textit{(komutativita)}$
	\item $(A+B)+C=A+(B+C) \hfill \textit{(asociativita)}$
	\item $A+\mathbf{0}=A \hfill \textit{(existence nulového prvku)}$
	\item $A+(-1)A=\mathbf{0} \hfill \textit{(existence opaèného prvku)}$
	\item $\alpha(\beta A)=(\alpha \beta)A$
	\item $1\cdot A=A$
	\item $\alpha(A+B)=\alpha A + \alpha B \hfill \textit{(distributivita)}$
	\item $(\alpha+\beta)A=\alpha A + \beta A \hfill \textit{(distributivita)}$
\end{penumerate}
Tedy prostor matic typu $m\times n$ odpovídá vektorovému prostoru.
\end{e}

\subsubsection*{Násobení}

\begin{e}{Definice}{0}{Maticové násobení}
Je-li A matice typu $m \times p$ a B matice typu $p \times n$, potom $A\cdot B$ je matice typu $m \times n$ definaná pøedpisem $$(A\cdot B)_{ij} = \sum_{k=1}^p A_{ik} B_{kj}$$ pro $i=1,\dots,m; j=1,\dots,n$.
\end{e}


\begin{e}{Lemma}{0}{Vlastnosti souèinu matic}

Nech $A,B,C$ jsou matice, $\alpha$ skalár. Potom
\begin{penumerate}
	\item Jestlie souèin $(AB)C$ je definován, potom i souèin $A(BC)$ je definován a platí $(AB)C=A(BC)$.
	\item Jestlie $A(B+C)$ je definován, potom i $AB+AC$ je definován a platí $A(B+C)=AB+AC$.
	\item Jestlie $(A+B)C$ je definován, potom i $AC+BC$ je definován a platí $(A+B)C=AC+BC$.
	\item Je-li $AB$ definován, je $\alpha (AB)=(\alpha A)B=A(\alpha B)$
	\item Je-li $A$ typu $m \times n$, potom $I_m A=A I_n =A$.
\end{penumerate}

\noindent Násobení matic není komutativní - tj. obecnì neplatí $AB=BA$.
\end{e}

\begin{e}{Vìta}{0}{O hodnosti souèinu matic}
Pro matici $A$ typu $m\times p$ a matici $B$ typu $p\times n$ platí:
$$\rank(AB)\leq \min\{\rank(A),\rank(B)\}$$

\begin{e}{Dùkaz}{0}{0}
Øádkovı prostor $AB$ je urèitì podprostorem øádkového prostoru matice $B$ a sloupcovı prostor $AB$ podprostorem sloupcového prostoru matice $A$.
\end{e}
\end{e}


\subsubsection*{Transpozice}


\begin{e}{Definice}{0}{0}
Pro matici $A \in \mathbb{R}^{m \times n}$ definujeme \emph{transponovanou matici} $A^T \in \mathbb{R}^{n \times m}$ pøedpisem $$(A^T)_{ji} = A_{ij} \, (i=1, \dots, m; j=1, \dots, n)$$
\end{e}

\begin{e}{Lemma}{0}{Vlastnosti transpozice}
\begin{penumerate}
	\item $(A^T)^T = A$
	\item jsou-li $A,B$ stejného typu, je $(A+B)^T = A^T + B^T$
	\item $(\alpha A)^T=\alpha A^T$, pro kadé $\alpha \in \mathbb{R}$
	\item je-li $AB$ definován, je i $B^T A^T$ definován a platí $(AB)^T = B^T A^T$.
\end{penumerate}
\end{e}

\begin{e}{Definice}{0}{Symetrická matice}
Matice A se nazıvá \emph{symetrická} jestlie $A^T=A$.
\end{e}

\begin{e}{Vìta}{0}{0}
Pro kadou matici $A \in \mathbb{R}^{m \times n}$ je $A^T A$ symetrická.
\end{e}
\begin{e}{Vìta}{0}{0}
Pro kadou matici $A \in \mathbb{R}^{m \times n}$ platí $\rank(A^T)=\rank(A)$.
\end{e}


\subsection{Inversní matice}

\begin{e}{Vìta}{0}{0}
Ke kadé regulární matici $A \in \mathbb{R}^{n \times n}$ existuje právì jedna matice $A^{-1} \in \mathbb{R}^{n \times n}$ s vlastností
$$A A^{-1} = A^{-1}A=I$$
Naopak, existuje-li k $A \in \mathbb{R}^{n \times n}$ matice $A^{-1}$ s touto vlastností, potom je A regulární.
\end{e}

\begin{e}{Definice}{0}{0}
Matici $A^{-1}$ s touto vlastností nazıváme \emph{inversní maticí} k matici A.
\end{e}

\begin{e}{Poznámka}{0}{0}
Inverzní matici mají tedy právì regulární matice.
\end{e}
\begin{e}{Dùsledek}{0}{0}
Je-li A regulární, je i $A^{-1}$ regulární.
\end{e}

\begin{e}{Vìta}{0}{Inversní matice je oboustrannì inversní}
Jestlie pro $A,X \in \mathbb{R}^{n \times n}$ platí $XA=I$, potom A je regulární a $X=A^{-1}$. Analogicky, jestlie $AX=I$, potom A je regulární a $X=A^{-1}$.
\end{e}

\begin{e}{Vìta}{0}{0}
Je-li $A \in \mathbb{R}^{n \times n}$ regulární, potom pro kadé $b \in \mathbb{R}^n$ je jediné øešení soustavy $Ax=b$ dáno vzorcem $x = A^{-1}b$.
\end{e}

\begin{e}{Vìta}{0}{Vıpoèet inversní matice}
Pro ètvercovou matici $A$ øádu $n$ nech je matice $(A\ I)$ (tj. zøetìzení sloupcù matice $A$ a jednotkové matice $I$ øádu $n$) pøevedena  Gauss-Jordanovou eliminací na tvar $(I\ X)$. Potom platí: 
$$X=A^{-1}$$ 
Jestlie Gauss-Jordanova eliminace není proveditelná a do konce, potom $A$ je singulární a nemá inversní matici.

\medskip
\begin{e}{Dùkaz}{0}{0}
Víme, e Gauss-Jordanova eliminace je vlastnì opakované násobení regulárními maticemi zleva. Souèin všech tìchto matic oznaème $Q$. Oznaème $H_{*,j}$ $j$-tı sloupec nìjaké (obecné) matice. Potom pro $j\in\{1,\dots,n\}$ platí: $(I\ X)_{*,j}=I_{*,j}=Q(A\ I)_{*,j}=(QA)_{*,j}$, tedy $QA=I$, dále platí $(I\ X)_{*,n+j}=X_{*,j}=(QI)_{*,n+j}=Q_{*,j}$, take $Q=X$ a tedy $AX=I$.
\end{e}
\end{e}

\begin{e}{Vìta}{0}{Vlastnosti inversní matice}
Nech $A,B \in \mathbb{R}^{n \times n}$ jsou regulární matice. Potom platí:
\begin{penumerate}
	\item $(A^{-1})^{-1} = A$
	\item $(A^{T})^{-1} = (A^{-1})^{T}$
	\item $(\alpha A)^{-1} = \frac{1}{\alpha} A^{-1} \textit{ pro }\alpha \neq 0$
	\item $(AB)^{-1} = B^{-1} A^{-1}$
\end{penumerate}
\end{e}

\subsection{Matice a lineární zobrazení, resp. zmìny souøadnıch soustav}

\begin{e}{Definice}{0}{0}
Nech V, W jsou vektorové prostory nad stejnım tìlesem (R nebo C). Zobrazení $f: V \rightarrow W$ nazıváme \emph{lineárním zobrazením} jestlie
\begin{penumerate}
	\item $f(x+y)=f(x)+f(y)$ pro kadé $x,y \in V$
	\item $f(\alpha\cdot x)=\alpha\cdot f(x)$ pro kadé $x \in V$ a kadı skalár $\alpha$.
\end{penumerate}
\end{e}

\begin{e}{Definice}{0}{Souøadnicovı vektor}
Nech $\mathbb{B}=(x_1,\dots,x_n)$ je báze V. Kadı vektor $x \in V$ lze potom vyjádøit právì jedním zpùsobem jako lineární kombinaci vektorù báze $\mathbb{B}$. Potom aritmetickı vektor
$$[x]_{\mathbb{B}} = \left( \begin{array}{l}\alpha_1 \\ \vdots \\ \alpha_n \end{array} \right)$$
nazıváme \emph{souøadnicovım vektorem} vektoru x v bázi $\mathbb{B}$ (a $n=\dim V$ a souøadnicovı vektor \emph{závisí na vıbìru báze}).
\end{e}


\begin{e}{Definice}{0}{Matice lineárního zobrazení}
Nech $\mathbb{B}=\{x_1, \dots, x_n\}$ je báze vektorového prostoru $V$, $\mathbb{B}'=\{y_1, \dots, y_m\}$ je báze vekt. prostrou $W$ a nech $f: V \rightarrow W$ je lineární zobrazení. Potom pro kadé $j=1, \dots, n$ lze $f(x_j)$ zapsat právì jedním zpùsobem ve tvaru
$$f(x_j) = \sum_{i = 1}^{m} \alpha_{ij} y_i.$$

Matice $A=(\alpha_{ij}) \in \mathbb{R}^{m \times n}$ se nazıvá maticí lineárního zobrazení $f$ vzhledem k bázím $\mathbb{B}, \mathbb{B}'$ a znaèí se
$$[f]_{\mathbb{B}\mathbb{B}'}.$$
\end{e}

\begin{e}{Pozorování}{0}{0}
$[f]_{\mathbb{B}\mathbb{B}'}$ je matice sestavená ze sloupcù
$$([f(x_1)]_{\mathbb{B}'}, \dots, [f(x_n)]_{\mathbb{B}'}),$$
které jsou souøadnicovımi vektory vektorù $f(x_1), \dots, f(x_n)$ v bázi $\mathbb{B}'$.
\end{e}

\begin{e}{Vìta}{0}{0}
Nech $\mathbb{B}$ je báze V, $\mathbb{B}'$ je báze W, a nech $f: V \rightarrow W$ je lineární zobrazení. Potom pro kadé $x \in V$ platí
$$[f(x)]_{\mathbb{B}'} = [f]_{\mathbb{B} \mathbb{B}'}\cdot[x]_{\mathbb{B}},$$
kde napravo stojí maticovı souèin.
\end{e}

\begin{e}{Vìta}{0}{Sloené zobrazení a maticovı souèin}
Nech $f: U \rightarrow V$, $g: V \rightarrow W$ jsou lineární zobrazení a nech $\mathbb{B}, \mathbb{B}', \mathbb{B}''$ jsou báze U, V, W. Potom platí
$$[g \circ f]_{\mathbb{B} \mathbb{B}''}=[g]_{\mathbb{B}'\mathbb{B}''}\cdot[f]_{\mathbb{B}\mathbb{B}'}$$
kde napravo stojí maticovı souèin.
\end{e}

\begin{e}{Vìta}{0}{Matice inversního zobrazení}
Je-li $f: V \rightarrow W$ isomorfismus, potom inversní zobrazení $f^{-1}: W \rightarrow V$ je rovnì isomorfismus a vzhledem k libovolnım bázím $\mathbb{B}, \mathbb{B}'$ prostorù V, W platí:
$$[f^{-1}]_{\mathbb{B}' \mathbb{B}} = [f]^{-1}_{\mathbb{B} \mathbb{B}'}$$
\end{e}

\begin{e}{Vìta}{0}{Zmìna souøadnic vektoru pøi zmìnì báze}
Nech jsou dány dvì báze $\mathbb{B}, \mathbb{B}'$ vektorového prostoru $V$. Potom pro kadé $x \in V$ platí:
$$[x]_{\mathbb{B}'} = [\mathrm{id}_V]_{\mathbb{B} \mathbb{B}'}\cdot[x]_{\mathbb{B}}$$

Matice $[\mathrm{id}_V]_{\mathbb{B} \mathbb{B}'}$ se nazıvá \emph{maticí pøechodu} od báze $\mathbb{B}$ k bázi $\mathbb{B}'$.
\end{e}

\begin{e}{Poznámka}{0}{0}
Pøedchozí vzorec vyaduje znalost hodnot vektorù staré báze $\mathbb{B}$ v nové bází $\mathbb{B}'$. Typická situace ale je, e máme jen starou bázi $\mathbb{B}$ a pomocí ní vyjádøíme novou bázi $\mathbb{B}'$. V tom pøípadì mùeme pouít vzorec
$$[x]_{\mathbb{B}'} = [\mathrm{id}_V]^{-1}_{\mathbb{B}' \mathbb{B}}\cdot[x]_{\mathbb{B}}$$
\end{e}
