\begin{e}{Definice}{0}{Dùleité typy grafù}

\item \textbf{úplnı graf $K_n$:} $V=\{1,\dots,n\}, E=\binom{V}{2}$ \textit{(Kadı vrchol je spojen hranou s kadım napø. $K_5$ --- \uv{pentagram}.)}

\item \textbf{bipartitní graf:} $V=\{u_1,\dots,u_n\} \; \cup \; \{v_1,\dots,v_m\}$, $E \subseteq \{\{u_i,v_j\}; i=1,\dots,n, j=1,\dots,m\}$; v \textbf{úplném bipartitním grafu} je $E = \{\{u_i,v_j\}; i=1,\dots,n, j=1,\dots,m\}$ \textit{(Kadı vrchol z jedné partity je spojen hranou pouze s nìkterımi (v úplném z kadım) vrcholem druhé partity. \subsection{Eulerovské a hamiltonovské grafy}

\begin{e}{Definice}{0}{Eulerovskı graf}

\medskip
\begin{e}{Dùkaz}{0}{0}
\uv{$\Rightarrow$}: tato implikace je triviální --- eulerovskı graf musí bıt souvislı, do kadého vrcholu musím vstoupit i z nìj vystoupit, co zvıší stupeò vdy o 2.

\bigskip
\noindent \uv{$\Leftarrow$}: Mìjme tah maximální délky $v_0,e_1,\dots,e_m,v_m.$ První a poslední vrchol tahu jsou totoné, jinak by do prvního vrcholu vedl lichı poèet hran, a jeliko graf má všechny stupnì sudé, dal by se tah prodouit. \end{e}
\end{e}
